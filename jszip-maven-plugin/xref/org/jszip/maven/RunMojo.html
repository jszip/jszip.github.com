211     public void execute()
212             throws MojoExecutionException, MojoFailureException {
213         if (runPackages == null || runPackages.length == 0) {
214             runPackages = new String[]{"war"};
215         }
216 
217         injectMissingArtifacts(project, executedProject);
218 
219         if (!Arrays.asList(runPackages).contains(project.getPackaging())) {
220             getLog().info("Skipping JSZip run: module " + ArtifactUtils.versionlessKey(project.getGroupId(),
221                     project.getArtifactId()) + " as not specified in runPackages");
222             return;
223         }
224         if (StringUtils.isNotBlank(runModule)
225                 && !project.getArtifactId().equals(runModule)
226                 && !ArtifactUtils.versionlessKey(project.getGroupId(), project.getArtifactId()).equals(runModule)) {
227             getLog().info("Skipping JSZip run: module " + ArtifactUtils.versionlessKey(project.getGroupId(),
228                     project.getArtifactId()) + " as requested runModule is " + runModule);
229             return;
230         }
231         getLog().info("Starting JSZip run: module " + ArtifactUtils.versionlessKey(project.getGroupId(),
232                 project.getArtifactId()));
233         MavenProject project = this.project;
234         long lastClassChange = System.currentTimeMillis();
235         long lastPomChange = getPomsLastModified();
236 
237         Server server = new Server();
238         if (connectors == null || connectors.length == 0) {
239             SelectChannelConnector selectChannelConnector = new SelectChannelConnector();
240             selectChannelConnector.setPort(8080);
241             connectors = new Connector[]{
242                     selectChannelConnector
243             };
244         }
245         server.setConnectors(connectors);
246         ContextHandlerCollection contexts = new ContextHandlerCollection();
247         HandlerCollection handlerCollection = new HandlerCollection(true);
248         DefaultHandler defaultHandler = new DefaultHandler();
249         handlerCollection.setHandlers(new Handler[]{contexts, defaultHandler});
250         server.setHandler(handlerCollection);
251         try {
252             server.start();
253         } catch (Exception e) {
254             throw new MojoExecutionException(e.getMessage(), e);
255         }
256         List<MavenProject> reactorProjects = this.reactorProjects;
257         WebAppContext webAppContext;
258         Resource webXml;
259         try {
260             List<Resource> resources = new ArrayList<Resource>();
261             for (Artifact a : getOverlayArtifacts(project, scope)) {
262                 MavenProject fromReactor = findProject(reactorProjects, a);
263                 if (fromReactor != null) {
264                     File jsDir = new File(fromReactor.getBasedir(), "src/main/js");
265                     if (jsDir.isDirectory()) {
266                         resources.add(Resource.newResource(jsDir));
267                     }
268                     // TODO filtering support
269                     File resDir = new File(fromReactor.getBasedir(), "src/main/resources");
270                     if (resDir.isDirectory()) {
271                         resources.add(Resource.newResource(resDir));
272                     }
273                 } else {
274                     resources.add(Resource.newResource("jar:" + a.getFile().toURI().toURL() + "!/"));
275                 }
276             }
277             if (webAppSourceDirectory == null) {
278                 webAppSourceDirectory = new File(project.getBasedir(), "src/main/webapp");
279             }
280             if (webAppSourceDirectory.isDirectory()) {
281                 resources.add(Resource.newResource(webAppSourceDirectory));
282             }
283             Collections.reverse(resources);
284             final ResourceCollection resourceCollection =
285                     new ResourceCollection(resources.toArray(new Resource[resources.size()]));
286 
287             webAppContext = new WebAppContext();
288             webAppContext.setWar(webAppSourceDirectory.getAbsolutePath());
289             webAppContext.setBaseResource(resourceCollection);
290 
291             WebAppClassLoader classLoader = new WebAppClassLoader(webAppContext);
292             for (String s : getClasspathElements(project, scope)) {
293                 classLoader.addClassPath(s);
294             }
295             webAppContext.setClassLoader(classLoader);
296 
297             contexts.setHandlers(new Handler[]{webAppContext});
298             contexts.start();
299             webAppContext.start();
300             Resource webInf = webAppContext.getWebInf();
301             webXml = webInf != null ? webInf.getResource("web.xml") : null;
302         } catch (ArtifactFilterException e) {
303             throw new MojoExecutionException(e.getMessage(), e);
304         } catch (MalformedURLException e) {
305             throw new MojoExecutionException(e.getMessage(), e);
306         } catch (IOException e) {
307             throw new MojoExecutionException(e.getMessage(), e);
308         } catch (Exception e) {
309             throw new MojoExecutionException(e.getMessage(), e);
310         }
311 
312         long webXmlLastModified = webXml == null ? 0L : webXml.lastModified();
313         try {
314 
315             getLog().info("Context started. Will restart if changes to poms detected.");
316             long nextClasspathCheck = System.currentTimeMillis() + classpathCheckInterval;
317             while (true) {
318                 long pomsLastModified = getPomsLastModified();
319                 boolean pomsChanged = lastPomChange < pomsLastModified;
320                 boolean overlaysChanged = false;
321                 boolean classPathChanged = webXmlLastModified < (webXml == null ? 0L : webXml.lastModified());
322                 if (nextClasspathCheck < System.currentTimeMillis()) {
323                     long classChange = classpathLastModified(project);
324                     if (classChange > lastClassChange) {
325                         classPathChanged = true;
326                         lastClassChange = classChange;
327                     }
328                     nextClasspathCheck = System.currentTimeMillis() + classpathCheckInterval;
329                 }
330                 if (!classPathChanged && !overlaysChanged && !pomsChanged) {
331                     try {
332                         Thread.sleep(500);
333                     } catch (InterruptedException e) {
334                         getLog().debug("Interrupted", e);
335                     }
336                     continue;
337                 }
338                 if (pomsChanged) {
339                     getLog().info("Change in poms detected, re-parsing to evaluate impact...");
340                     // we will now process this change,
341                     // so from now on don't re-process
342                     // even if we have issues processing
343                     lastPomChange = pomsLastModified;
344                     List<MavenProject> newReactorProjects;
345                     try {
346                         newReactorProjects = buildReactorProjects();
347                     } catch (ProjectBuildingException e) {
348                         getLog().error(e.getLocalizedMessage(), e);
349                         getLog().info("Re-parse aborted due to malformed pom.xml file(s)");
350                         continue;
351                     } catch (CycleDetectedException e) {
352                         getLog().error(e.getLocalizedMessage(), e);
353                         getLog().info("Re-parse aborted due to dependency cycle in project model");
354                         continue;
355                     } catch (DuplicateProjectException e) {
356                         getLog().error(e.getLocalizedMessage(), e);
357                         getLog().info("Re-parse aborted due to duplicate projects in project model");
358                         continue;
359                     } catch (Exception e) {
360                         getLog().error(e.getLocalizedMessage(), e);
361                         getLog().info("Re-parse aborted due a problem that prevented sorting the project model");
362                         continue;
363                     }
364                     if (!buildPlanEqual(newReactorProjects, this.reactorProjects)) {
365                         throw new BuildPlanModifiedException("A pom.xml change has impacted the build plan.");
366                     }
367                     MavenProject newProject = findProject(newReactorProjects, this.project);
368                     if (newProject == null) {
369                         throw new BuildPlanModifiedException(
370                                 "A pom.xml change appears to have removed " + this.project.getId()
371                                         + " from the build plan.");
372                     }
373 
374                     newProject.setArtifacts(resolve(newProject, "runtime"));
375 
376                     getLog().info("Comparing effective classpath of new and old models");
377                     try {
378                         classPathChanged = classPathChanged || classpathsEqual(project, newProject, scope);
379                     } catch (DependencyResolutionRequiredException e) {
380                         getLog().error(e.getLocalizedMessage(), e);
381                         getLog().info("Re-parse aborted due to dependency resolution problems");
382                         continue;
383                     }
384                     if (!classPathChanged) {
385                         if (classPathChanged) {
386                             getLog().info("Effective classpath of " + project.getId() + " has changed.");
387                         } else {
388                             getLog().info("Effective classpath is unchanged.");
389                         }
390                     }
391 
392                     getLog().debug("Comparing effective overlays of new and old models");
393                     try {
394                         overlaysChanged = overlaysEqual(project, newProject);
395                     } catch (OverConstrainedVersionException e) {
396                         getLog().error(e.getLocalizedMessage(), e);
397                         getLog().info("Re-parse aborted due to dependency resolution problems");
398                         continue;
399                     } catch (ArtifactFilterException e) {
400                         getLog().error(e.getLocalizedMessage(), e);
401                         getLog().info("Re-parse aborted due to overlay resolution problems");
402                         continue;
403                     }
404 
405                     project = newProject;
406                     reactorProjects = newReactorProjects;
407                 }
408 
409                 if (!overlaysChanged && !classPathChanged) {
410                     continue;
411                 }
412                 getLog().info("Restarting context to take account of changes...");
413                 try {
414                     webAppContext.stop();
415                 } catch (Exception e) {
416                     throw new MojoExecutionException(e.getMessage(), e);
417                 }
418 
419                 if (classPathChanged) {
420                     getLog().info("Updating classpath...");
421                     try {
422                         WebAppClassLoader classLoader = new WebAppClassLoader(webAppContext);
423                         for (String s : getClasspathElements(project, scope)) {
424                             classLoader.addClassPath(s);
425                         }
426                         webAppContext.setClassLoader(classLoader);
427                     } catch (Exception e) {
428                         throw new MojoExecutionException(e.getMessage(), e);
429                     }
430                 }
431 
432                 if (overlaysChanged) {
433                     getLog().info("Updating overlays...");
434                     try {
435                         List<Resource> resources = new ArrayList<Resource>();
436                         for (Artifact a : getOverlayArtifacts(project, scope)) {
437                             MavenProject fromReactor = findProject(reactorProjects, a);
438                             if (fromReactor != null) {
439                                 File jsDir = new File(fromReactor.getBasedir(), "src/main/js");
440                                 if (jsDir.isDirectory()) {
441                                     resources.add(Resource.newResource(jsDir));
442                                 }
443                                 File resDir = new File(fromReactor.getBasedir(), "src/main/resources");
444                                 if (resDir.isDirectory()) {
445                                     resources.add(Resource.newResource(resDir));
446                                 }
447                             } else {
448                                 resources.add(Resource.newResource("jar:" + a.getFile().toURI().toURL() + "!/"));
449                             }
450                         }
451                         if (webAppSourceDirectory.isDirectory()) {
452                             resources.add(Resource.newResource(webAppSourceDirectory));
453                         }
454                         Collections.reverse(resources);
455                         final ResourceCollection resourceCollection =
456                                 new ResourceCollection(resources.toArray(new Resource[resources.size()]));
457 
458                         webAppContext.setBaseResource(resourceCollection);
459                     } catch (Exception e) {
460                         throw new MojoExecutionException(e.getMessage(), e);
461                     }
462                 }
463                 try {
464                     webAppContext.start();
465                 } catch (Exception e) {
466                     throw new MojoExecutionException(e.getMessage(), e);
467                 }
468                 webXmlLastModified = webXml == null ? 0L : webXml.lastModified();
469                 getLog().info("Context restarted.");
470             }
471 
472         } finally {
473             try {
474                 server.stop();
475             } catch (Exception e) {
476                 throw new MojoExecutionException(e.getMessage(), e);
477             }
478         }
479     }

481     private void injectMissingArtifacts(MavenProject destination, MavenProject source) {
482         if (destination.getArtifact().getFile() == null && source.getArtifact().getFile() != null) {
483             getLog().info("Pushing primary artifact from forked execution into current execution");
484             destination.getArtifact().setFile(source.getArtifact().getFile());
485         }
486         for (Artifact executedArtifact : source.getAttachedArtifacts()) {
487             String executedArtifactId =
488                     (executedArtifact.getClassifier() == null ? "." : "-" + executedArtifact.getClassifier() + ".")
489                             + executedArtifact.getType();
490             if (StringUtils.equals(executedArtifact.getGroupId(), destination.getGroupId())
491                     && StringUtils.equals(executedArtifact.getArtifactId(), destination.getArtifactId())
492                     && StringUtils.equals(executedArtifact.getVersion(), destination.getVersion())) {
493                 boolean found = false;
494                 for (Artifact artifact : destination.getAttachedArtifacts()) {
495                     if (StringUtils.equals(artifact.getGroupId(), destination.getGroupId())
496                             && StringUtils.equals(artifact.getArtifactId(), destination.getArtifactId())
497                             && StringUtils.equals(artifact.getVersion(), destination.getVersion())
498                             && StringUtils.equals(artifact.getClassifier(), executedArtifact.getClassifier())
499                             && StringUtils.equals(artifact.getType(), executedArtifact.getType())) {
500                         if (artifact.getFile() == null) {
501                             getLog().info("Pushing " + executedArtifactId
502                                     + " artifact from forked execution into current execution");
503                             artifact.setFile(executedArtifact.getFile());
504                         }
505                         found = true;
506                     }
507                 }
508                 if (!found) {
509                     getLog().info("Attaching " +
510                             executedArtifactId
511                             + " artifact from forked execution into current execution");
512                     projectHelper
513                             .attachArtifact(destination, executedArtifact.getType(), executedArtifact.getClassifier(),
514                                     executedArtifact.getFile());
515                 }
516             }
517         }
518     }

520     private List<MavenProject> buildReactorProjects() throws Exception {
521 
522         List<MavenProject> projects = new ArrayList<MavenProject>();
523         for (MavenProject p : reactorProjects) {
524             ProjectBuildingRequest request = new DefaultProjectBuildingRequest();
525 
526             request.setProcessPlugins(true);
527             request.setProfiles(request.getProfiles());
528             request.setActiveProfileIds(session.getRequest().getActiveProfiles());
529             request.setInactiveProfileIds(session.getRequest().getInactiveProfiles());
530             request.setRemoteRepositories(session.getRequest().getRemoteRepositories());
531             request.setSystemProperties(session.getSystemProperties());
532             request.setUserProperties(session.getUserProperties());
533             request.setRemoteRepositories(session.getRequest().getRemoteRepositories());
534             request.setPluginArtifactRepositories(session.getRequest().getPluginArtifactRepositories());
535             request.setRepositorySession(session.getRepositorySession());
536             request.setLocalRepository(localRepository);
537             request.setBuildStartTime(session.getRequest().getStartTime());
538             request.setResolveDependencies(true);
539             request.setValidationLevel(ModelBuildingRequest.VALIDATION_LEVEL_STRICT);
540             projects.add(projectBuilder.build(p.getFile(), request).getProject());
541         }
542         return new ProjectSorter(projects).getSortedProjects();
543     }

545     private long classpathLastModified(MavenProject project) {
546         long result = Long.MIN_VALUE;
547         try {
548             for (String element : getClasspathElements(project, scope)) {
549                 File elementFile = new File(element);
550                 if (elementFile.exists()) {
551                     result = Math.max(elementFile.lastModified(), result);
552                     if (elementFile.isDirectory()) {
553                         Stack<Iterator<File>> stack = new Stack<Iterator<File>>();
554                         stack.push(contentsAsList(elementFile).iterator());
555                         while (!stack.empty()) {
556                             Iterator<File> i = stack.pop();
557                             while (i.hasNext()) {
558                                 File file = i.next();
559                                 result = Math.max(file.lastModified(), result);
560                                 if (file.isDirectory()) {
561                                     stack.push(i);
562                                     i = contentsAsList(file).iterator();
563                                 }
564                             }
565                         }
566                     }
567                 }
568             }
569         } catch (DependencyResolutionRequiredException e) {
570             // ignore
571         }
572         return result;
573 
574     }

576     private static List<File> contentsAsList(File directory) {
577         File[] files = directory.listFiles();
578         return files == null ? Collections.<File>emptyList() : Arrays.asList(files);
579     }

581     private boolean classpathsEqual(MavenProject oldProject, MavenProject newProject, String scope)
582             throws DependencyResolutionRequiredException {
583         int seq = 0;
584         List<String> newCP = getClasspathElements(newProject, scope);
585         List<String> oldCP = getClasspathElements(oldProject, scope);
586         boolean classPathChanged = newCP.size() != oldCP.size();
587         for (Iterator<String> i = newCP.iterator(), j = oldCP.iterator(); i.hasNext() || j.hasNext(); ) {
588             String left = i.hasNext() ? i.next() : "(empty)";
589             String right = j.hasNext() ? j.next() : "(empty)";
590             if (!StringUtils.equals(left, right)) {
591                 getLog().debug("classpath[" + seq + "]");
592                 getLog().debug("  old = " + left);
593                 getLog().debug("  new = " + right);
594                 classPathChanged = true;
595             }
596             seq++;
597         }
598         return classPathChanged;
599     }

601     private MavenProject findProject(List<MavenProject> newReactorProjects, MavenProject oldProject) {
602         final String targetId = oldProject.getId();
603         for (MavenProject newProject : newReactorProjects) {
604             if (targetId.equals(newProject.getId())) {
605                 return newProject;
606             }
607         }
608         return null;
609     }

611     private MavenProject findProject(List<MavenProject> projects, Artifact artifact) {
612         for (MavenProject project : projects) {
613             if (StringUtils.equals(artifact.getGroupId(), project.getGroupId())
614                     && StringUtils.equals(artifact.getArtifactId(), project.getArtifactId())
615                     && StringUtils.equals(artifact.getVersion(), project.getVersion())) {
616                 return project;
617             }
618         }
619         return null;
620     }

622     private boolean buildPlanEqual(List<MavenProject> newPlan, List<MavenProject> oldPlan) {
623         if (newPlan.size() != oldPlan.size()) {
624             return false;
625         }
626         int seq = 0;
627         for (Iterator<MavenProject> i = newPlan.iterator(), j = oldPlan.iterator(); i.hasNext() && j.hasNext(); ) {
628             MavenProject left = i.next();
629             MavenProject right = j.next();
630             getLog().debug(
631                     "[" + (seq++) + "] = " + left.equals(right) + (left == right ? " same" : " diff") + " : "
632                             + left.getName() + "[" + left.getDependencies().size() + "], " + right.getName()
633                             + "["
634                             + right.getDependencies().size() + "]");
635             if (!left.equals(right)) {
636                 return false;
637             }
638             if (left.getDependencies().size() != right.getDependencies().size()) {
639                 getLog().info("Dependency tree of " + left.getId() + " has been modified");
640             }
641         }
642         return true;
643     }

645     private boolean overlaysEqual(MavenProject oldProject, MavenProject newProject)
646             throws ArtifactFilterException, OverConstrainedVersionException {
647         boolean overlaysChanged;
648         Set<Artifact> newOA = getOverlayArtifacts(newProject, scope);
649         Set<Artifact> oldOA = getOverlayArtifacts(oldProject, scope);
650         overlaysChanged = newOA.size() != oldOA.size();
651         for (Artifact n : newOA) {
652             boolean found = false;
653             for (Artifact o : oldOA) {
654                 if (StringUtils.equals(n.getArtifactId(), o.getArtifactId()) && StringUtils
655                         .equals(n.getGroupId(), o.getGroupId())) {
656                     if (o.getSelectedVersion().equals(n.getSelectedVersion())) {
657                         found = true;
658                         break;
659                     }
660                 }
661             }
662             if (!found) {
663                 getLog().debug("added overlay artifact: " + n);
664                 overlaysChanged = true;
665             }
666         }
667         for (Artifact o : oldOA) {
668             boolean found = false;
669             for (Artifact n : newOA) {
670                 if (StringUtils.equals(n.getArtifactId(), o.getArtifactId()) && StringUtils
671                         .equals(n.getGroupId(), o.getGroupId())) {
672                     if (o.getSelectedVersion().equals(n.getSelectedVersion())) {
673                         found = true;
674                         break;
675                     }
676                 }
677             }
678             if (!found) {
679                 getLog().debug("removed overlay artifact: " + o);
680                 overlaysChanged = true;
681             }
682         }
683         if (overlaysChanged) {
684             getLog().info("Effective overlays of " + oldProject.getId() + " have changed.");
685         } else {
686             getLog().debug("Effective overlays are unchanged.");
687         }
688         return overlaysChanged;
689     }
