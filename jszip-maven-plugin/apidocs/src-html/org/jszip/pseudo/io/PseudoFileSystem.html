019    import org.apache.commons.lang3.StringUtils;
020    import org.codehaus.plexus.archiver.zip.ZipEntry;
021    import org.codehaus.plexus.archiver.zip.ZipFile;
022    import org.mozilla.javascript.Context;
023    
024    import java.io.File;
025    import java.io.IOException;
026    import java.util.ArrayList;
027    import java.util.Arrays;
028    import java.util.Collections;
029    import java.util.Enumeration;
030    import java.util.LinkedHashSet;
031    import java.util.List;
032    import java.util.Map;
033    import java.util.Set;
034    import java.util.TreeMap;

036    public class PseudoFileSystem {
037        /**
038         * Secret key used to hold the reference to the pseudo filesystem.
039         */
040        private static final Object KEY = new Object();
041    
042        private final PseudoFile root = new VirtualDirectoryPseudoFile(null, "");
043    
044        private final Layer[] layers;
045    
046        public PseudoFileSystem(Layer... layers) {
047            this.layers = layers;
048        }
049    
050        public String getPathSeparator() {
051            return "/";
052        }
053    
054        public PseudoFile[] listChildren(PseudoFile dir, PseudoFileFilter filter) {
055            TreeMap<String, Layer> names = new TreeMap<String, Layer>();
056            final String path = dir.getAbsolutePath();
057            for (int i = layers.length - 1; i >= 0; i--) {
058                for (String name : layers[i].listChildren(path)) {
059                    names.put(name, layers[i]);
060                }
061            }
062            List<PseudoFile> result = new ArrayList<PseudoFile>(names.size());
063            for (Map.Entry<String, Layer> entry : names.entrySet()) {
064                if (filter.accept(entry.getKey())) {
065                    result.add(entry.getValue().makeChild(dir, entry.getKey()));
066                }
067            }
068            return result.toArray(new PseudoFile[result.size()]);
069        }
070    
071        public PseudoFile root() {
072            return root;
073        }
074    
075        public PseudoFile getPseudoFile(String filename) {
076            filename = StringUtils.removeEnd(filename, getPathSeparator());
077            if (filename.isEmpty()) {
078                return root();
079            }
080            int index = filename.lastIndexOf(getPathSeparator());
081            if (index != -1) {
082                return getPseudoFile(getPseudoFile(filename.substring(0, index)), filename.substring(index + 1));
083            }
084            return getPseudoFile(root(), filename);
085        }
086    
087        public PseudoFile getPseudoFile(PseudoFile parent, String name) {
088            if (name.equals(".")) {
089                return parent;
090            }
091            if (name.equals("..")) {
092                return parent.getParentFile();
093            }
094            String parentPath = parent.getAbsolutePath();
095            for (Layer layer : layers) {
096                if (layer.listChildren(parentPath).contains(name)) {
097                    return layer.makeChild(parent, name);
098                }
099            }
100            if (layers.length == 0) {
101                return new VirtualDirectoryPseudoFile(parent, name);
102            }
103            return layers[0].makeChild(parent, name);
104        }
105    
106        public synchronized void installInContext() {
107            Context.getCurrentContext().putThreadLocal(KEY, this);
108        }
109    
110        public synchronized void removeFromContext() {
111            Context.getCurrentContext().putThreadLocal(KEY, null);
112        }
113    
114        public static PseudoFileSystem current() {
115            return (PseudoFileSystem) Context.getCurrentContext().getThreadLocal(KEY);
116        }
117    
118        public abstract static class Layer {
119    
120            public abstract List<String> listChildren(String relativePath);
121    
122            public abstract PseudoFile makeChild(PseudoFile parent, String name);
123    
124        }
125    
126        public static class FileLayer extends Layer {
127            private final String prefix;
128            private final File root;
129    
130            public FileLayer(File root) {
131                this("", root);
132            }
133    
134            public FileLayer(String prefix, File root) {
135                this.prefix = StringUtils.isEmpty(prefix) ? "/" : "/" + StringUtils.removeEnd(
136                        StringUtils.removeStart(prefix, "/"), "/") + "/";
137                this.root = root;
138            }
139    
140            @Override
141            public List<String> listChildren(String relativePath) {
142                relativePath = StringUtils.removeEnd(relativePath, "/") + "/";
143                if (relativePath.startsWith(prefix)) {
144                    final String pathFragment = relativePath.substring(prefix.length());
145                    final String[] list = new File(root, pathFragment).list();
146                    return list == null ? Collections.<String>emptyList() : Arrays.asList(list);
147                }
148                if (prefix.startsWith(relativePath)) {
149                    int index = prefix.indexOf('/', relativePath.length() + 1);
150                    if (index != -1) {
151                        return Collections.singletonList(prefix.substring(relativePath.length(), index));
152                    }
153                }
154                return Collections.emptyList();
155            }
156    
157            @Override
158            public PseudoFile makeChild(PseudoFile parent, String name) {
159                String relativePath = StringUtils.removeEnd(parent.getAbsolutePath(), "/") + "/" + name;
160                if (relativePath.startsWith(prefix)) {
161                    return new FilePseudoFile(parent, new File(root, relativePath.substring(prefix.length())));
162                }
163                if (prefix.equals(relativePath + "/")) {
164                    int lastIndex = prefix.lastIndexOf('/');
165                    int index = prefix.lastIndexOf('/', lastIndex - 1);
166                    return new AliasFilePseudoFile(parent, root, prefix.substring(index + 1, lastIndex));
167                }
168                if (!StringUtils.isEmpty(prefix) && prefix.startsWith(relativePath)) {
169                    return new VirtualDirectoryPseudoFile(parent, name);
170                }
171                return new NotExistingPseudoFile(parent, name);
172            }
173    
174            @Override
175            public String toString() {
176                final StringBuilder sb = new StringBuilder();
177                sb.append("FileLayer");
178                sb.append("{prefix='").append(prefix).append('\'');
179                sb.append(", root=").append(root);
180                sb.append('}');
181                return sb.toString();
182            }
183        }
184    
185        public static class ZipLayer extends PseudoFileSystem.Layer {
186            private final String prefix;
187            private final File zipFile;
188            private final Map<String, ZipEntry> contents;
189    
190            public ZipLayer(String prefix, File zipFile) throws IOException {
191                this.prefix = StringUtils.isEmpty(prefix) ? "/" : "/" + StringUtils.removeEnd(
192                        StringUtils.removeStart(prefix, "/"), "/") + "/";
193                this.zipFile = zipFile;
194                ZipFile file = new ZipFile(zipFile);
195                Map<String, ZipEntry> contents = new TreeMap<String, ZipEntry>();
196                Enumeration<ZipEntry> entries = file.getEntries();
197                while (entries.hasMoreElements()) {
198                    ZipEntry entry = entries.nextElement();
199                    contents.put(this.prefix + StringUtils.removeStart(entry.getName(), "/"), entry);
200                }
201    
202                this.contents = contents;
203            }
204    
205            @Override
206            public List<String> listChildren(String relativePath) {
207                relativePath = StringUtils.removeEnd(relativePath, "/") + "/";
208                if (relativePath.startsWith(prefix) || prefix.equals(relativePath + "/")) {
209                    final String pathFragment = StringUtils.removeEnd(relativePath,"/") + "/";
210                    Set<String> result = new LinkedHashSet<String>();
211                    for (String path : contents.keySet()) {
212                        if (path.startsWith(pathFragment)) {
213                            int index = path.indexOf('/', pathFragment.length());
214                            if (index == -1) {
215                                result.add(path.substring(pathFragment.length()));
216                            } else {
217                                result.add(path.substring(pathFragment.length(), index));
218                            }
219                        }
220                    }
221                    return new ArrayList<String>(result);
222                }
223                if (prefix.startsWith(relativePath)) {
224                    int index = prefix.indexOf('/', relativePath.length());
225                    if (index != -1) {
226                        return Collections.singletonList(prefix.substring(relativePath.length(), index));
227                    }
228                }
229                return Collections.emptyList();
230            }
231    
232            @Override
233            public PseudoFile makeChild(PseudoFile parent, String name) {
234                String relativePath = StringUtils.removeEnd(parent.getAbsolutePath(), "/") + "/" + name;
235                final ZipEntry entry = contents.get(relativePath);
236                if (entry != null) {
237                    return new ZipPseudoFile(parent, zipFile, entry);
238                }
239                if (prefix.equals(relativePath + "/")) {
240                    return new VirtualDirectoryPseudoFile(parent, name);
241                }
242                if (!StringUtils.isEmpty(prefix) && prefix.startsWith(relativePath)) {
243                    return new VirtualDirectoryPseudoFile(parent, name);
244                }
245                for (String childPath: contents.keySet()) {
246                    if (!StringUtils.isEmpty(childPath) && childPath.startsWith(relativePath)) {
247                        return new VirtualDirectoryPseudoFile(parent, name);
248                    }
249                }
250                return new NotExistingPseudoFile(parent, name);
251            }
252    
253            @Override
254            public String toString() {
255                final StringBuilder sb = new StringBuilder();
256                sb.append("ZipLayer");
257                sb.append("{prefix='").append(prefix).append('\'');
258                sb.append(", zipFile=").append(zipFile);
259                sb.append('}');
260                return sb.toString();
261            }
262        }
