041    /**
042     * JettyWebAppContext
043     *
044     * Extends the WebAppContext to specialize for the maven environment.
045     * We pass in the list of files that should form the classpath for
046     * the webapp when executing in the plugin, and any jetty-env.xml file
047     * that may have been configured.
048     *
049     */
050    public class JettyWebAppContext extends WebAppContext
051    {
052        private static final String WEB_INF_CLASSES_PREFIX = "/WEB-INF/classes";
053        private static final String WEB_INF_LIB_PREFIX = "/WEB-INF/lib";
054    
055        private final List<File> webInfClasses = new ArrayList<File>();
056        private final List<File> webInfJars = new ArrayList<File>();
057        private final Map<String, File> webInfJarMap = new HashMap<String, File>();
058        private final EnvConfiguration envConfig;
059        private List<File> classpathFiles;
060        private String jettyEnvXml;
061        private List<Resource> overlays;
062        private boolean unpackOverlays;
063        private String containerIncludeJarPattern = ".*/servlet-api-[^/]*\\.jar$";
064    
065        public JettyWebAppContext ()
066        throws Exception
067        {
068            super();
069            setConfigurations(new Configuration[]{
070                    new WebXmlConfiguration(),
071                    new MetaInfConfiguration(),
072                    new FragmentConfiguration(),
073                    envConfig = new EnvConfiguration(),
074                    new org.eclipse.jetty.plus.webapp.PlusConfiguration(),
075                    new JettyWebXmlConfiguration(),
076            });
077        }

144        /* ------------------------------------------------------------ */
145        /**
146         * This method is provided as a conveniance for jetty maven plugin configuration
147         * @param resourceBases Array of resources strings to set as a {@link ResourceCollection}. Each resource string may be a comma separated list of resources
148         * @see Resource
149         */
150        public void setResourceBases(String[] resourceBases)
151        {
152            List<String> resources = new ArrayList<String>();
153            for (String rl:resourceBases)
154            {
155                String[] rs = rl.split(" *, *");
156                for (String r:rs)
157                    resources.add(r);
158            }
159    
160            setBaseResource(new ResourceCollection(resources.toArray(new String[resources.size()])));
161        }
162    
163        public List<File> getWebInfLib()
164        {
165            return webInfJars;
166        }
167    
168        public void doStart () throws Exception
169        {
170            setAttribute(WebInfConfiguration.CONTAINER_JAR_PATTERN, containerIncludeJarPattern);
171    
172    
173            // Initialize map containing all jars in /WEB-INF/lib
174            webInfJarMap.clear();
175            for (File file : webInfJars)
176            {
177                // Return all jar files from class path
178                String fileName = file.getName();
179                if (fileName.endsWith(".jar"))
180                    webInfJarMap.put(fileName, file);
181            }
182    
183            if (this.jettyEnvXml != null)
184                envConfig.setJettyEnvXml(new File(this.jettyEnvXml).toURL());
185            setShutdown(false);
186            super.doStart();
187        }
188    
189        public void doStop () throws Exception
190        {
191            setShutdown(true);
192            //just wait a little while to ensure no requests are still being processed
193            Thread.currentThread().sleep(500L);
194            super.doStop();
195        }

197        @Override
198        public Resource getResource(String uriInContext) throws MalformedURLException
199        {
200            Resource resource = null;
201            // Try to get regular resource
202            resource = super.getResource(uriInContext);
203    
204            // If no regular resource exists check for access to /WEB-INF/lib or /WEB-INF/classes
205            if ((resource == null || !resource.exists()) && uriInContext != null && webInfClasses != null)
206            {
207                String uri = URIUtil.canonicalPath(uriInContext);
208    
209                try
210                {
211                    // Replace /WEB-INF/classes with real classes directory
212                    if (uri.startsWith(WEB_INF_CLASSES_PREFIX))
213                    {
214                        Resource res = null;
215                        int i=0;
216                        while (res == null && (i < webInfClasses.size()))
217                        {
218                            String newPath = uri.replace(WEB_INF_CLASSES_PREFIX, webInfClasses.get(i).getPath());
219                            res = Resource.newResource(newPath);
220                            if (!res.exists())
221                            {
222                                res = null;
223                                i++;
224                            }
225                        }
226                        return res;
227                    }
228                    // Return the real jar file for all accesses to
229                    // /WEB-INF/lib/*.jar
230                    else if (uri.startsWith(WEB_INF_LIB_PREFIX))
231                    {
232                        String jarName = uri.replace(WEB_INF_LIB_PREFIX, "");
233                        if (jarName.startsWith("/") || jarName.startsWith("\\"))
234                            jarName = jarName.substring(1);
235                        if (jarName.length()==0)
236                            return null;
237                        File jarFile = webInfJarMap.get(jarName);
238                        if (jarFile != null)
239                            return Resource.newResource(jarFile.getPath());
240    
241                        return null;
242                    }
243                }
244                catch (MalformedURLException e)
245                {
246                    throw e;
247                }
248                catch (IOException e)
249                {
250                    Log.ignore(e);
251                }
252            }
253            return resource;
254        }

256        @Override
257        public Set<String> getResourcePaths(String path)
258        {
259            // Try to get regular resource paths
260            Set<String> paths = super.getResourcePaths(path);
261    
262            // If no paths are returned check for virtual paths /WEB-INF/classes and /WEB-INF/lib
263            if (paths.isEmpty() && path != null)
264            {
265                path = URIUtil.canonicalPath(path);
266                if (path.startsWith(WEB_INF_LIB_PREFIX))
267                {
268                    paths = new TreeSet<String>();
269                    for (String fileName : webInfJarMap.keySet())
270                    {
271                        // Return all jar files from class path
272                        paths.add(WEB_INF_LIB_PREFIX + "/" + fileName);
273                    }
274                }
275                else if (path.startsWith(WEB_INF_CLASSES_PREFIX))
276                {
277                    int i=0;
278    
279                    while (paths.isEmpty() && (i < webInfClasses.size()))
280                    {
281                        String newPath = path.replace(WEB_INF_CLASSES_PREFIX, webInfClasses.get(i).getPath());
282                        paths = super.getResourcePaths(newPath);
283                        i++;
284                    }
285                }
286            }
287            return paths;
288        }
