001    package org.jszip.jetty;
002    
003    import org.eclipse.jetty.util.URIUtil;
004    import org.eclipse.jetty.util.resource.Resource;
005    
006    import java.io.File;
007    import java.io.FileNotFoundException;
008    import java.io.IOException;
009    import java.io.InputStream;
010    import java.io.OutputStream;
011    import java.net.MalformedURLException;
012    import java.net.URL;
013    
014    /**
015     * @author stephenc
016     * @since 03/12/2012 09:18
017     */
018    public class VirtualDirectoryResource extends Resource {
019    
020        private Resource child;
021        private final String name;
022    
023        public VirtualDirectoryResource(Resource child, String name) {
024            int index = name.indexOf(URIUtil.SLASH);
025            if (index == 0) {
026                name = name.substring(1);
027                index = name.indexOf(URIUtil.SLASH);
028            }
029            if ((index == -1) || (index == name.length() - 1)) {
030                this.child = child;
031                this.name = name;
032            } else {
033                this.child = new VirtualDirectoryResource(child, name.substring(index + 1));
034                this.name = name.substring(0, index);
035            }
036        }
037    
038        public Resource getChild() {
039            return child;
040        }
041    
042        public void setChild(Resource child) {
043            this.child = child;
044        }
045    
046        @Override
047        public boolean isContainedIn(Resource resource) throws MalformedURLException {
048            return false;
049        }
050    
051        @Override
052        public void release() {
053            if (child != null) {
054                child.release();
055            }
056        }
057    
058        @Override
059        public boolean exists() {
060            return true;
061        }
062    
063        @Override
064        public boolean isDirectory() {
065            return true;
066        }
067    
068        @Override
069        public long lastModified() {
070            return child != null ? child.lastModified() : -1;
071        }
072    
073        @Override
074        public long length() {
075            return -1;
076        }
077    
078        @Override
079        public URL getURL() {
080            // TODO return a non-null URL to allow directory browsing support
081            return null;
082        }
083    
084        @Override
085        public File getFile() throws IOException {
086            return null;
087        }
088    
089        @Override
090        public String getName() {
091            return null;
092        }
093    
094        @Override
095        public InputStream getInputStream() throws IOException {
096            return null;
097        }
098    
099        @Override
100        public OutputStream getOutputStream() throws IOException, SecurityException {
101            return null;
102        }
103    
104        @Override
105        public boolean delete() throws SecurityException {
106            throw new UnsupportedOperationException();
107        }
108    
109        @Override
110        public boolean renameTo(Resource resource) throws SecurityException {
111            throw new UnsupportedOperationException();
112        }
113    
114        @Override
115        public String[] list() {
116            return new String[]{name + "/"};
117        }
118    
119        @Override
120        public Resource addPath(String path) throws IOException, MalformedURLException {
121            if (path == null) {
122                throw new MalformedURLException();
123            }
124            if (path.length() == 0 || URIUtil.SLASH.equals(path)) {
125                return this;
126            }
127            if (path.startsWith(name) || path.startsWith(URIUtil.SLASH + name)) {
128                return child.addPath(path.substring(path.indexOf(name) + name.length()));
129            }
130            return new BadResource();
131        }
132    
133        @Override
134        public String toString() {
135            final StringBuilder sb = new StringBuilder();
136            sb.append("VirtualDirectoryResource");
137            sb.append("{name='").append(name).append('\'');
138            sb.append(", child=").append(child);
139            sb.append('}');
140            return sb.toString();
141        }
142    
143        @Override
144        public boolean equals(Object o) {
145            if (this == o) {
146                return true;
147            }
148            if (!(o instanceof VirtualDirectoryResource)) {
149                return false;
150            }
151    
152            VirtualDirectoryResource that = (VirtualDirectoryResource) o;
153    
154            if (child != null ? !child.equals(that.child) : that.child != null) {
155                return false;
156            }
157            if (name != null ? !name.equals(that.name) : that.name != null) {
158                return false;
159            }
160    
161            return true;
162        }
163    
164        @Override
165        public int hashCode() {
166            int result = child != null ? child.hashCode() : 0;
167            result = 31 * result + (name != null ? name.hashCode() : 0);
168            return result;
169        }
170    
171        class BadResource extends Resource {
172    
173            BadResource() {
174            }
175    
176    
177            @Override
178            public boolean isContainedIn(Resource r) throws MalformedURLException {
179                return false;
180            }
181    
182            @Override
183            public void release() {
184            }
185    
186            @Override
187            public boolean exists() {
188                return false;
189            }
190    
191            @Override
192            public long lastModified() {
193                return -1;
194            }
195    
196            @Override
197            public boolean isDirectory() {
198                return false;
199            }
200    
201            @Override
202            public long length() {
203                return -1;
204            }
205    
206            @Override
207            public URL getURL() {
208                return null;
209            }
210    
211            @Override
212            public File getFile() {
213                return null;
214            }
215    
216            @Override
217            public String getName() {
218                return null;
219            }
220    
221            @Override
222            public InputStream getInputStream() throws IOException {
223                throw new FileNotFoundException();
224            }
225    
226            @Override
227            public OutputStream getOutputStream()
228                    throws java.io.IOException, SecurityException {
229                throw new FileNotFoundException();
230            }
231    
232            @Override
233            public boolean delete() throws SecurityException {
234                throw new SecurityException();
235            }
236    
237            @Override
238            public boolean renameTo(Resource dest) throws SecurityException {
239                throw new SecurityException();
240            }
241    
242            @Override
243            public String[] list() {
244                return null;
245            }
246    
247            @Override
248            public Resource addPath(String path) throws IOException, MalformedURLException {
249                return this;
250            }
251    
252            @Override
253            public void copyTo(File destination) throws IOException {
254                throw new SecurityException();
255            }
256    
257            @Override
258            public String toString() {
259                return super.toString() + "; BadResource";
260            }
261    
262        }
