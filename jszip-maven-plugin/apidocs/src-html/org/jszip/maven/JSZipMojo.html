053    /**
054     * @phase package
055     * @goal jszip
056     * @requiresDependencyResolution runtime
057     */
058    public class JSZipMojo extends AbstractMojo {
059    
060        /**
061         * The maven project.
062         *
063         * @parameter expression="${project}"
064         * @required
065         * @readonly
066         */
067        private MavenProject project;
068    
069    
070        /**
071         * Directory containing the classes.
072         *
073         * @parameter expression="src/main/js"
074         * @required
075         */
076        private File contentDirectory;
077    
078        /**
079         * Directory containing the generated ZIP.
080         *
081         * @parameter expression="${project.build.directory}"
082         * @required
083         */
084        private File outputDirectory;
085    
086        /**
087         * Name of the generated ZIP.
088         *
089         * @parameter expression="${zip.finalName}" default-value="${project.build.finalName}"
090         * @required
091         */
092        private String finalName;
093    
094        /**
095         * Classifier to add to the artifact generated. If given, the artifact will be an attachment instead.
096         *
097         * @parameter
098         */
099        private String classifier;
100    
101        /**
102         * The Jar archiver.
103         *
104         * @component role="org.codehaus.plexus.archiver.Archiver" roleHint="zip"
105         */
106        private ZipArchiver zipArchiver;
107    
108        /**
109         * Include or not empty directories
110         *
111         * @parameter expression="${zip.includeEmptyDirs}" default-value="false"
112         */
113        private boolean includeEmptyDirs;
114    
115        /**
116         * Whether creating the archive should be forced.
117         *
118         * @parameter expression="${zip.forceCreation}" default-value="false"
119         */
120        private boolean forceCreation;
121    
122        /**
123         * Adding pom.xml and pom.properties to the archive.
124         *
125         * @parameter expression="${addMavenDescriptor}" default-value="true"
126         */
127        private boolean addMavenDescriptor;
128    
129    
130        protected File getZipFile(File basedir, String finalName, String classifier) {
131            if (classifier == null) {
132                classifier = "";
133            } else if (classifier.trim().length() > 0 && !classifier.startsWith("-")) {
134                classifier = "-" + classifier;
135            }
136    
137            return new File(basedir, finalName + classifier + ".zip");
138        }
139    
140        /**
141         * @see org.apache.maven.plugin.Mojo#execute()
142         */
143        public void execute()
144                throws MojoExecutionException, MojoFailureException {
145            try {
146    
147                File zipFile = getZipFile(outputDirectory, finalName, classifier);
148    
149                zipArchiver.setDestFile(zipFile);
150                zipArchiver.setIncludeEmptyDirs(includeEmptyDirs);
151                zipArchiver.setCompress(true);
152                zipArchiver.setForced(forceCreation);
153    
154                if (addMavenDescriptor) {
155                    if (project.getArtifact().isSnapshot()) {
156                        project.setVersion(project.getArtifact().getVersion());
157                    }
158    
159                    String groupId = project.getGroupId();
160    
161                    String artifactId = project.getArtifactId();
162    
163                    zipArchiver.addFile(project.getFile(), "META-INF/maven/" + groupId + "/" + artifactId + "/pom.xml");
164                    zipArchiver.addResource(new PomPropertiesResource(project),
165                            "META-INF/maven/" + groupId + "/" + artifactId + "/pom.properties",
166                            zipArchiver.getOverrideFileMode());
167                }
168                zipArchiver.addResource(new PackageJsonResource(project), "package.json",
169                        zipArchiver.getOverrideFileMode());
170                if (contentDirectory.isDirectory()) {
171                    zipArchiver.addDirectory(contentDirectory);
172                }
173                zipArchiver.createArchive();
174                project.getArtifact().setFile(zipFile);
175    
176            } catch (Exception e) {
177                throw new MojoExecutionException("Error assembling ZIP", e);
178            }
179    
180        }

182        private static class PackageJsonResource extends AbstractPlexusIoResource {
183            private final byte[] bytes;
184            private final long lastModified;
185    
186            public PackageJsonResource(MavenProject project) throws IOException, MojoExecutionException {
187                this.lastModified = project.getFile().lastModified();
188                Map<String, Object> p = new TreeMap<String, Object>();
189                p.put("name", project.getGroupId() + "." + project.getArtifactId());
190                p.put("version", project.getVersion());
191                addFirstNotEmpty(p, "description", project.getDescription());
192                addFirstNotEmpty(p, "homepage", project.getUrl());
193                if (project.getDevelopers() != null && !project.getDevelopers().isEmpty()) {
194                    List<Object> devs = new ArrayList<Object>();
195                    for (Developer d : (List<Developer>) project.getDevelopers()) {
196                        Map<String, Object> dev = new TreeMap<String, Object>();
197                        addFirstNotEmpty(dev, "name", d.getName(), d.getId(), d.getEmail());
198                        addFirstNotEmpty(dev, "email", d.getEmail());
199                        addFirstNotEmpty(dev, "web", d.getUrl());
200                        if (dev.containsKey("name")) {
201                            devs.add(dev);
202                        }
203                    }
204                    p.put("maintainers", devs);
205                }
206                if (project.getContributors() != null && !project.getContributors().isEmpty()) {
207                    List<Object> contribs = new ArrayList<Object>();
208                    for (Contributor c : (List<Contributor>) project.getContributors()) {
209                        Map<String, Object> contrib = new TreeMap<String, Object>();
210                        addFirstNotEmpty(contrib, "name", c.getName(), c.getEmail());
211                        addFirstNotEmpty(contrib, "email", c.getEmail());
212                        addFirstNotEmpty(contrib, "web", c.getUrl());
213                        if (contrib.containsKey("name")) {
214                            contribs.add(contrib);
215                        }
216                    }
217                    p.put("contributors", contribs);
218                }
219                if (project.getIssueManagement() != null) {
220                    addFirstNotEmpty(p, "bugs", project.getIssueManagement().getUrl());
221                }
222                if (project.getLicenses() != null && !project.getLicenses().isEmpty()) {
223                    List<Object> licenses = new ArrayList<Object>();
224                    for (License l : (List<License>) project.getLicenses()) {
225                        Map<String, Object> license = new TreeMap<String, Object>();
226                        addFirstNotEmpty(license, "type", l.getName());
227                        addFirstNotEmpty(license, "url", l.getUrl());
228                        licenses.add(license);
229                    }
230                    p.put("licenses", licenses);
231    
232                }
233                FilterArtifacts filter = new FilterArtifacts();
234        
235                filter.addFilter(new ProjectTransitivityFilter(project.getDependencyArtifacts(), true));
236        
237                filter.addFilter(new ScopeFilter("runtime", ""));
238        
239                filter.addFilter(new TypeFilter("jszip", ""));
240        
241                // start with all artifacts.
242                Set<Artifact> artifacts = project.getArtifacts();
243        
244                // perform filtering
245                try {
246                    artifacts = filter.filter(artifacts);
247                } catch (ArtifactFilterException e) {
248                    throw new MojoExecutionException(e.getMessage(), e);
249                }
250    
251                Map<String,String> dependencies = new LinkedHashMap<String, String>();
252                for (Artifact artifact : artifacts) {
253                    dependencies.put(artifact.getGroupId() + "." + artifact.getArtifactId(), artifact.getVersion());
254                }
255                p.put("dependencies", dependencies);
256        
257                ByteArrayOutputStream os = new ByteArrayOutputStream();
258                try {
259                    ObjectMapper m = new ObjectMapper();
260                    m.setSerializationConfig(m.getSerializationConfig().with(SerializationConfig.Feature.INDENT_OUTPUT));
261                    m.writeValue(os, p);
262                } finally {
263                    IOUtil.close(os);
264                }
265                bytes = os.toByteArray();
266    
267            }
268    
269            private void addFirstNotEmpty(Map<String, Object> map, String name, String... values) {
270                for (String value : values) {
271                    if (StringUtils.isNotEmpty(value)) {
272                        map.put(name, value);
273                        return;
274                    }
275                }
276            }
277    
278            @Override
279            public boolean isFile() {
280                return true;
281            }
282    
283            @Override
284            public boolean isExisting() {
285                return true;
286            }
287    
288            @Override
289            public long getLastModified() {
290                return lastModified;
291            }
292    
293            @Override
294            public long getSize() {
295                return bytes.length;
296            }
297    
298            public URL getURL() throws IOException {
299                return null;
300            }
301    
302            public InputStream getContents() throws IOException {
303                return new ByteArrayInputStream(bytes);
304            }
305        }

307        private static class PomPropertiesResource extends AbstractPlexusIoResource {
308            private static final String GENERATED_BY_MAVEN = "Generated by Maven";
309            private final byte[] bytes;
310            private final MavenProject project;
311    
312            public PomPropertiesResource(MavenProject project) throws IOException {
313                this.project = project;
314    
315                Properties p = new Properties();
316    
317                p.setProperty("groupId", project.getGroupId());
318    
319                p.setProperty("artifactId", project.getArtifactId());
320    
321                p.setProperty("version", project.getVersion());
322    
323                ByteArrayOutputStream os = new ByteArrayOutputStream();
324                try {
325                    p.store(os, GENERATED_BY_MAVEN);
326                } finally {
327                    IOUtil.close(os);
328                }
329    
330                bytes = os.toByteArray();
331            }
332    
333            @Override
334            public boolean isFile() {
335                return true;
336            }
337    
338            @Override
339            public boolean isExisting() {
340                return true;
341            }
342    
343            @Override
344            public long getLastModified() {
345                return project.getFile().lastModified();
346            }
347    
348            @Override
349            public long getSize() {
350                return bytes.length;
351            }
352    
353            public URL getURL() throws IOException {
354                return null;
355            }
356    
357            public InputStream getContents() throws IOException {
358                return new ByteArrayInputStream(bytes);
359            }
