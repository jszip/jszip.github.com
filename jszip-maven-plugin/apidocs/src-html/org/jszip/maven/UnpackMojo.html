054    /**
055     * Unpacks all the JSZip dependencies into a web application.
056     *
057     * @phase generate-resources
058     * @goal unpack
059     * @requiresDependencyResolution compile+runtime
060     */
061    public class UnpackMojo extends AbstractJSZipMojo {
062    
063        /**
064         * The directory where the webapp is built.
065         *
066         * @parameter expression="${project.build.directory}/${project.build.finalName}"
067         * @required
068         */
069        private File webappDirectory;
070    
071        /**
072         * The Zip unarchiver.
073         *
074         * @component role="org.codehaus.plexus.archiver.UnArchiver" roleHint="zip"
075         */
076        private ZipUnArchiver zipUnArchiver;
077    
078        /**
079         * @parameter expression="${reactorProjects}"
080         * @required
081         * @readonly
082         */
083        protected List<MavenProject> reactorProjects;
084    
085        /**
086         * The Maven plugin Manager
087         *
088         * @component
089         * @readonly
090         * @required
091         */
092        private MavenPluginManager mavenPluginManager;
093    
094        /**
095         * The current build session instance. This is used for plugin manager API calls.
096         *
097         * @parameter expression="${session}"
098         * @required
099         * @readonly
100         */
101        private MavenSession session;
102    
103        /**
104         * @parameter expression="${plugin}"
105         * @readonly
106         */
107        private PluginDescriptor pluginDescriptor;
108    
109        /**
110         * @see org.apache.maven.plugin.Mojo#execute()
111         */
112        public void execute()
113                throws MojoExecutionException, MojoFailureException {
114            getLog().info("Starting unpack into " + webappDirectory);
115            FilterArtifacts filter = new FilterArtifacts();
116    
117            filter.addFilter(new ProjectTransitivityFilter(project.getDependencyArtifacts(), false));
118    
119            filter.addFilter(new ScopeFilter("runtime", ""));
120    
121            filter.addFilter(new TypeFilter(JSZIP_TYPE, ""));
122    
123            // start with all artifacts.
124            Set<Artifact> artifacts = project.getArtifacts();
125    
126            // perform filtering
127            try {
128                artifacts = filter.filter(artifacts);
129            } catch (ArtifactFilterException e) {
130                throw new MojoExecutionException(e.getMessage(), e);
131            }
132    
133            for (Artifact artifact : artifacts) {
134                getLog().info("Unpacking " + ArtifactUtils.key(artifact));
135                unpack(artifact, webappDirectory, null, null);
136            }
137        }
138    
139        protected void unpack(Artifact artifact, File location, String includes, String excludes)
140                throws MojoExecutionException {
141            File file = artifact.getFile();
142            if (file.isDirectory()) {
143                MavenProject fromReactor = findProject(reactorProjects, artifact);
144                if (fromReactor != null) {
145                    MavenSession session = this.session.clone();
146                    session.setCurrentProject(fromReactor);
147                    Plugin plugin = findThisPluginInProject(fromReactor);
148                    try {
149                        // we cheat here and use our version of the plugin... but this is less of a cheat than the only
150                        // other way which is via reflection.
151                        MojoDescriptor jszipDescriptor = findMojoDescriptor(this.pluginDescriptor, JSZipMojo.class);
152    
153                        for (PluginExecution pluginExecution : plugin.getExecutions()) {
154                            if (!pluginExecution.getGoals().contains(jszipDescriptor.getGoal())) {
155                                continue;
156                            }
157                            MojoExecution mojoExecution =
158                                    createMojoExecution(plugin, pluginExecution, jszipDescriptor);
159                            JSZipMojo mojo = (JSZipMojo)mavenPluginManager
160                                    .getConfiguredMojo(Mojo.class, session, mojoExecution);
161                            try {
162                                File contentDirectory = mojo.getContentDirectory();
163                                if (contentDirectory.isDirectory()) {
164                                    FileUtils.copyDirectory(contentDirectory, location);
165                                }
166                                File resourcesDirectory = mojo.getResourcesDirectory();
167                                if (resourcesDirectory.isDirectory()) {
168                                    FileUtils.copyDirectory(resourcesDirectory, location);
169                                }
170                            } finally {
171                                mavenPluginManager.releaseMojo(mojo, mojoExecution);
172                            }
173                        }
174                    } catch (PluginConfigurationException e) {
175                        throw new MojoExecutionException(e.getMessage(), e);
176                    } catch (IOException e) {
177                        throw new MojoExecutionException(e.getMessage(), e);
178                    } catch (PluginContainerException e) {
179                        throw new MojoExecutionException(e.getMessage(), e);
180                    }
181                } else {
182                    throw new MojoExecutionException("Cannot find jzsip artifact: " + artifact.getId());
183                }
184            } else {
185                try {
186                    location.mkdirs();
187    
188                    zipUnArchiver.setSourceFile(file);
189    
190                    zipUnArchiver.setDestDirectory(location);
191    
192                    if (StringUtils.isNotEmpty(excludes) || StringUtils.isNotEmpty(includes)) {
193                        IncludeExcludeFileSelector[] selectors =
194                                new IncludeExcludeFileSelector[]{new IncludeExcludeFileSelector()};
195    
196                        if (StringUtils.isNotEmpty(excludes)) {
197                            selectors[0].setExcludes(excludes.split(","));
198                        }
199    
200                        if (StringUtils.isNotEmpty(includes)) {
201                            selectors[0].setIncludes(includes.split(","));
202                        }
203    
204                        zipUnArchiver.setFileSelectors(selectors);
205                    }
206    
207                    zipUnArchiver.extract();
208                } catch (ArchiverException e) {
209                    e.printStackTrace();
210                    throw new MojoExecutionException("Error unpacking file: " + file + " to: " + location + "\r\n"
211                            + e.toString(), e);
212                }
213            }
214        }
