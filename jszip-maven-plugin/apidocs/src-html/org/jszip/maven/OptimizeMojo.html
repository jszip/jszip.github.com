068    /**
069     * Runs the r.js optimizer over the source
070     */
071    @Mojo(name = "optimize", defaultPhase = LifecyclePhase.PROCESS_CLASSES,
072            requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME)
073    public class OptimizeMojo extends AbstractJSZipMojo {

075        /**
076         * Regex for a quoted string which may include escapes.
077         */
078        public static final String QUOTED_STRING_WITH_ESCAPES = "'([^\\\\']+|\\\\([btnfr\"'\\\\]|[0-3]?[0-7]{1,"
079                + "2}|u[0-9a-fA-F]{4}))*'|\"([^\\\\\"]+|\\\\([btnfr\"'\\\\]|[0-3]?[0-7]{1,2}|u[0-9a-fA-F]{4}))*\"";

081        /**
082         * Regex for sniffing the version of r.js being used.
083         */
084        public static final String R_JS_VERSION_REGEX = "\\s+version\\s*=\\s*(" + QUOTED_STRING_WITH_ESCAPES + ")";

160        /**
161         * @see org.apache.maven.plugin.Mojo#execute()
162         */
163        public void execute() throws MojoExecutionException, MojoFailureException {
164            if (skip) {
165                getLog().info("Optimization skipped.");
166                return;
167            }
168            if (!contentDirectory.exists()) {
169                getLog().info("Nothing to do, no r.js build profiles in " + contentDirectory);
170                return;
171            }
172            if (!contentDirectory.isDirectory()) {
173                throw new MojoExecutionException("Build profile directory '" + contentDirectory + "' is not a directory");
174            }
175            if (webappDirectory.isFile()) {
176                throw new MojoExecutionException("Webapp directory '" + webappDirectory + "' is not a directory");
177            }
178            if (!webappDirectory.isDirectory() && !webappDirectory.mkdirs()) {
179                throw new MojoExecutionException("Could not create Webapp directory '" + webappDirectory + "'");
180            }
181            String source;
182            int lineNo = 0;
183            InputStream inputStream = null;
184            InputStreamReader reader = null;
185            try {
186                if (customRScript.isFile()) {
187                    getLog().debug("Using custom r.js from: " + customRScript);
188                    inputStream = new FileInputStream(customRScript);
189                } else {
190                    getLog().debug("Using bundled r.js");
191                    inputStream = getClass().getResourceAsStream("/org/jszip/maven/r.js");
192                }
193                source = IOUtil.toString(inputStream, "UTF-8");
194                if (source.startsWith("#!")) {
195                    int i1 = source.indexOf('\n');
196                    int i2 = source.indexOf('\r');
197                    int index = (i1 == -1 || i2 == -1) ? Math.max(i1, i2) : Math.min(i1, i2);
198                    if (index > 0) {
199                        source = source.substring(index);
200                        lineNo++;
201                    }
202                }
203            } catch (IOException e) {
204                throw new MojoExecutionException(e.getMessage(), e);
205            } finally {
206                IOUtil.close(reader);
207                IOUtil.close(inputStream);
208            }
209    
210            String sourceVersion = "unknown";
211            Pattern rJsVersionPattern = Pattern.compile(R_JS_VERSION_REGEX);
212            Matcher rJsVersionMatcher = rJsVersionPattern.matcher(source);
213            if (rJsVersionMatcher.find()) {
214                sourceVersion = rJsVersionMatcher.group(1);
215            }
216    
217            getLog().info("Using r.js version " + sourceVersion);
218    
219            List<PseudoFileSystem.Layer> layers = buildVirtualFileSystemLayers();
220    
221            final ContextFactory contextFactory = new ShellContextFactory();
222            final Global global = new Global();
223            global.initQuitAction(new QuitAction() {
224                public void quit(Context context, int exitCode) {
225                    if (exitCode != 0) {
226                        throw new JavaScriptTerminationException("Script exited with exit code of " + exitCode, exitCode);
227                    }
228                }
229            });
230            if (!global.isInitialized()) {
231                global.init(contextFactory);
232            }
233            DirectoryScanner scanner = new DirectoryScanner();
234    
235            scanner.setBasedir(contentDirectory);
236    
237            if (includes != null && !includes.isEmpty()) {
238                scanner.setIncludes(processIncludesExcludes(includes));
239            } else {
240                scanner.setIncludes(new String[]{"**/*.js"});
241            }
242    
243            if (excludes != null && !excludes.isEmpty()) {
244                scanner.setExcludes(processIncludesExcludes(excludes));
245            } else {
246                scanner.setExcludes(new String[]{"r.js"});
247            }
248    
249            scanner.scan();
250    
251            for (String path : scanner.getIncludedFiles()) {
252                File profileJs = new File(contentDirectory, path);
253                PseudoFileSystem.Layer[] layersArray = layers.toArray(new PseudoFileSystem.Layer[layers.size() + 1]);
254                layersArray[layers.size()] = new PseudoFileSystem.FileLayer("build", profileJs.getParentFile());
255                try {
256                    Object rv = contextFactory
257                            .call(new OptimizeContextAction(getLog(), global, profileJs, source, lineNo, layersArray));
258                    if (rv instanceof Number) {
259                        if (((Number) rv).intValue() != 0) {
260                            throw new MojoExecutionException(
261                                    "Non-zero exit code of " + ((Number) rv).intValue()
262                                            + " when trying to optimize profile " + profileJs);
263                        }
264                    }
265                } catch (JavaScriptException e) {
266                    throw new MojoExecutionException(
267                            "Uncaught exception when trying to optimize profile " + profileJs, e);
268                } catch (JavaScriptTerminationException e) {
269                    throw new MojoExecutionException(
270                            "Non-zero exit code of " + e.getExitCode() + " when trying to optimize profile " + profileJs);
271                }
272            }
273        }
274    
275        private String getPath(Artifact artifact) {
276            if (mappings == null) {
277                return "/virtual";
278            }
279            for (Mapping mapping: mappings) {
280                if (mapping.isMatch(artifact)) {
281                    final String path = StringUtils.clean(mapping.getPath());
282                    if (StringUtils.isBlank(path) || "/".equals(path)) {
283                        return "/virtual";
284                    }
285                    return "/virtual/" + StringUtils.strip(path, "/");
286                }
287            }
288            return "/virtual";
289        }

291        private List<PseudoFileSystem.Layer> buildVirtualFileSystemLayers() throws MojoExecutionException {
292            List<PseudoFileSystem.Layer> layers = new ArrayList<PseudoFileSystem.Layer>();
293            layers.add(new PseudoFileSystem.FileLayer("/target", webappDirectory));
294            layers.add(new PseudoFileSystem.FileLayer("/virtual", warSourceDirectory));
295            FilterArtifacts filter = new FilterArtifacts();
296    
297            filter.addFilter(new ProjectTransitivityFilter(project.getDependencyArtifacts(), false));
298    
299            filter.addFilter(new ScopeFilter("runtime", ""));
300    
301            filter.addFilter(new TypeFilter(JSZIP_TYPE, ""));
302    
303            // start with all artifacts.
304            Set<Artifact> artifacts = project.getArtifacts();
305    
306            // perform filtering
307            try {
308                artifacts = filter.filter(artifacts);
309            } catch (ArtifactFilterException e) {
310                throw new MojoExecutionException(e.getMessage(), e);
311            }
312    
313            for (Artifact artifact : artifacts) {
314                String path = getPath(artifact);
315                getLog().info("Adding " + ArtifactUtils.key(artifact) + " to virtual filesystem");
316                File file = artifact.getFile();
317                if (file.isDirectory()) {
318                    MavenProject fromReactor = findProject(reactorProjects, artifact);
319                    if (fromReactor != null) {
320                        MavenSession session = this.session.clone();
321                        session.setCurrentProject(fromReactor);
322                        Plugin plugin = findThisPluginInProject(fromReactor);
323                        try {
324                            // we cheat here and use our version of the plugin... but this is less of a cheat than the only
325                            // other way which is via reflection.
326                            MojoDescriptor jszipDescriptor = findMojoDescriptor(this.pluginDescriptor, JSZipMojo.class);
327    
328                            for (PluginExecution pluginExecution : plugin.getExecutions()) {
329                                if (!pluginExecution.getGoals().contains(jszipDescriptor.getGoal())) {
330                                    continue;
331                                }
332                                MojoExecution mojoExecution =
333                                        createMojoExecution(plugin, pluginExecution, jszipDescriptor);
334                                JSZipMojo mojo = (JSZipMojo) mavenPluginManager
335                                        .getConfiguredMojo(org.apache.maven.plugin.Mojo.class, session, mojoExecution);
336                                try {
337                                    File contentDirectory = mojo.getContentDirectory();
338                                    if (contentDirectory.isDirectory()) {
339                                        getLog().debug("Merging directory " + contentDirectory + " into " + path);
340                                        layers.add(new PseudoFileSystem.FileLayer(path, contentDirectory));
341                                    }
342                                    File resourcesDirectory = mojo.getResourcesDirectory();
343                                    if (resourcesDirectory.isDirectory()) {
344                                        getLog().debug("Merging directory " + contentDirectory + " into " + path);
345                                        layers.add(new PseudoFileSystem.FileLayer(path, resourcesDirectory));
346                                    }
347                                } finally {
348                                    mavenPluginManager.releaseMojo(mojo, mojoExecution);
349                                }
350                            }
351                        } catch (PluginConfigurationException e) {
352                            throw new MojoExecutionException(e.getMessage(), e);
353                        } catch (PluginContainerException e) {
354                            throw new MojoExecutionException(e.getMessage(), e);
355                        }
356                    } else {
357                        throw new MojoExecutionException("Cannot find jzsip artifact: " + artifact.getId());
358                    }
359                } else {
360                    try {
361                        getLog().debug("Merging .zip file " + file + " into " + path);
362                        layers.add(new PseudoFileSystem.ZipLayer(path, file));
363                    } catch (IOException e) {
364                        throw new MojoExecutionException(e.getMessage(), e);
365                    }
366                }
367            }
368            return layers;
369        }
370    
371        private static String[] processIncludesExcludes(List<String> list) {
372            List<String> result = new ArrayList<String>();
373            for (String entry : list) {
374                String[] entries = entry.split(",");
375                Collections.addAll(result, entries);
376            }
377            return result.toArray(new String[result.size()]);
378        }
