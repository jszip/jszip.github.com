017    package org.jszip.maven;
018    
019    import org.apache.maven.artifact.Artifact;
020    import org.apache.maven.model.Plugin;
021    import org.apache.maven.model.PluginExecution;
022    import org.apache.maven.plugin.AbstractMojo;
023    import org.apache.maven.plugin.Mojo;
024    import org.apache.maven.plugin.MojoExecution;
025    import org.apache.maven.plugin.MojoExecutionException;
026    import org.apache.maven.plugin.descriptor.MojoDescriptor;
027    import org.apache.maven.plugin.descriptor.PluginDescriptor;
028    import org.apache.maven.project.MavenProject;
029    import org.codehaus.plexus.configuration.PlexusConfiguration;
030    import org.codehaus.plexus.util.StringUtils;
031    import org.codehaus.plexus.util.xml.Xpp3Dom;
032    import org.codehaus.plexus.util.xml.Xpp3DomUtils;
033    
034    import java.lang.reflect.InvocationTargetException;
035    import java.lang.reflect.Method;
036    import java.util.List;
037    
038    /**
039     * Common base class for all the JSZip goals.
040     */
041    public abstract class AbstractJSZipMojo extends AbstractMojo {
042        /**
043         * The type of packaging.
044         */
045        public static final String JSZIP_TYPE = "jszip";
046    
047        /**
048         * The maven project.
049         *
050         * @parameter expression="${project}"
051         * @required
052         * @readonly
053         */
054        protected MavenProject project;
055        /**
056         * The current plugin.
057         *
058         * @parameter expression="${plugin.groupId}"
059         * @required
060         * @readonly
061         */
062        private String pluginGroupId;
063        /**
064         * The current plugin.
065         *
066         * @parameter expression="${plugin.artifactId}"
067         * @required
068         * @readonly
069         */
070        private String pluginArtifactId;
071        /**
072         * The current plugin.
073         *
074         * @parameter expression="${plugin.version}"
075         * @required
076         * @readonly
077         */
078        private String pluginVersion;
079    
080        protected static <T> T invokeMethod(Object object, Class<T> rvClass, String method, Object... args)
081                throws MojoExecutionException {
082            Class[] argClasses = new Class[args.length];
083            for (int i = 0; i < args.length; i++) {
084                argClasses[i] = args.getClass();
085            }
086            try {
087                Method m = object.getClass().getMethod(method, argClasses);
088                return rvClass.cast(m.invoke(object, args));
089            } catch (InvocationTargetException e) {
090                throw new MojoExecutionException(e.getMessage(), e);
091            } catch (NoSuchMethodException e) {
092                throw new MojoExecutionException(e.getMessage(), e);
093            } catch (IllegalAccessException e) {
094                throw new MojoExecutionException(e.getMessage(), e);
095            }
096        }
097    
098    
099        protected MojoExecution createMojoExecution(Plugin plugin, PluginExecution pluginExecution,
100                                                    MojoDescriptor mojoDescriptor) {
101            MojoExecution mojoExecution = new MojoExecution(plugin, mojoDescriptor.getGoal(), pluginExecution.getId());
102            mojoExecution.setConfiguration(convert(mojoDescriptor));
103            if (plugin.getConfiguration() != null || pluginExecution.getConfiguration() != null) {
104                Xpp3Dom pluginConfiguration =
105                        plugin.getConfiguration() == null ? new Xpp3Dom("fake")
106                                : (Xpp3Dom) plugin.getConfiguration();
107    
108                Xpp3Dom mergedConfigurationWithExecution =
109                        Xpp3DomUtils.mergeXpp3Dom(
110                                (Xpp3Dom) pluginExecution.getConfiguration(),
111                                pluginConfiguration);
112    
113                Xpp3Dom mergedConfiguration =
114                        Xpp3DomUtils.mergeXpp3Dom(mergedConfigurationWithExecution,
115                                convert(mojoDescriptor));
116    
117                Xpp3Dom cleanedConfiguration = new Xpp3Dom("configuration");
118                if (mergedConfiguration.getChildren() != null) {
119                    for (Xpp3Dom parameter : mergedConfiguration.getChildren()) {
120                        if (mojoDescriptor.getParameterMap().containsKey(parameter.getName())) {
121                            cleanedConfiguration.addChild(parameter);
122                        }
123                    }
124                }
125                if (getLog().isDebugEnabled()) {
126                    getLog().debug("mojoExecution mergedConfiguration: " + mergedConfiguration);
127                    getLog().debug("mojoExecution cleanedConfiguration: " + cleanedConfiguration);
128                }
129    
130                mojoExecution.setConfiguration(cleanedConfiguration);
131    
132            }
133            mojoExecution.setMojoDescriptor(mojoDescriptor);
134            return mojoExecution;
135        }
136    
137        protected MojoDescriptor findMojoDescriptor(PluginDescriptor pluginDescriptor, Class<? extends Mojo> mojoClass) {
138            MojoDescriptor mojoDescriptor = null;
139            for (MojoDescriptor d : pluginDescriptor.getMojos()) {
140                if (mojoClass.getName().equals(d.getImplementation())) {
141                    mojoDescriptor = pluginDescriptor.getMojo(d.getGoal());
142                    break;
143                }
144            }
145    
146            if (mojoDescriptor == null) {
147                getLog().error("Cannot find goal that corresponds to " + mojoClass);
148                throw new IllegalStateException("This plugin should always have the " + mojoClass.getName() + " goal");
149            }
150            return mojoDescriptor;
151        }
152    
153        protected Plugin findThisPluginInProject(MavenProject project) {
154            Plugin plugin = null;
155            for (Plugin b : project.getBuild().getPlugins()) {
156                if (pluginGroupId.equals(b.getGroupId()) && pluginArtifactId.equals(b.getArtifactId())) {
157                    plugin = b.clone();
158                    plugin.setVersion(pluginVersion); // we want to use our version
159                    break;
160                }
161            }
162            if (plugin == null) {
163                getLog().debug("Falling back to our own plugin");
164                plugin = new Plugin();
165                plugin.setGroupId(pluginGroupId);
166                plugin.setArtifactId(pluginArtifactId);
167                plugin.setVersion(pluginVersion);
168            }
169            return plugin;
170        }
171    
172        private Xpp3Dom convert(MojoDescriptor mojoDescriptor) {
173            PlexusConfiguration config = mojoDescriptor.getMojoConfiguration();
174            return (config != null) ? convert(config) : new Xpp3Dom("configuration");
175        }
176    
177        private Xpp3Dom convert(PlexusConfiguration config) {
178            if (config == null) {
179                return null;
180            }
181    
182            Xpp3Dom dom = new Xpp3Dom(config.getName());
183            dom.setValue(config.getValue(null));
184    
185            for (String attrib : config.getAttributeNames()) {
186                dom.setAttribute(attrib, config.getAttribute(attrib, null));
187            }
188    
189            for (int n = config.getChildCount(), i = 0; i < n; i++) {
190                dom.addChild(convert(config.getChild(i)));
191            }
192    
193            return dom;
194        }
195    
196        protected MavenProject findProject(List<MavenProject> projects, Artifact artifact) {
197            for (MavenProject project : projects) {
198                if (StringUtils.equals(artifact.getGroupId(), project.getGroupId())
199                        && StringUtils.equals(artifact.getArtifactId(), project.getArtifactId())
200                        && StringUtils.equals(artifact.getVersion(), project.getVersion())) {
201                    return project;
202                }
203            }
204            return null;
205        }
206    }
