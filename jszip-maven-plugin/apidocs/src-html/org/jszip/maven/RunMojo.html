251        public void execute()
252                throws MojoExecutionException, MojoFailureException {
253            if (runPackages == null || runPackages.length == 0) {
254                runPackages = new String[]{"war"};
255            }
256    
257            injectMissingArtifacts(project, executedProject);
258    
259            if (!Arrays.asList(runPackages).contains(project.getPackaging())) {
260                getLog().info("Skipping JSZip run: module " + ArtifactUtils.versionlessKey(project.getGroupId(),
261                        project.getArtifactId()) + " as not specified in runPackages");
262                return;
263            }
264            if (StringUtils.isNotBlank(runModule)
265                    && !project.getArtifactId().equals(runModule)
266                    && !ArtifactUtils.versionlessKey(project.getGroupId(), project.getArtifactId()).equals(runModule)) {
267                getLog().info("Skipping JSZip run: module " + ArtifactUtils.versionlessKey(project.getGroupId(),
268                        project.getArtifactId()) + " as requested runModule is " + runModule);
269                return;
270            }
271            getLog().info("Starting JSZip run: module " + ArtifactUtils.versionlessKey(project.getGroupId(),
272                    project.getArtifactId()));
273            MavenProject project = this.project;
274            long lastResourceChange = System.currentTimeMillis();
275            long lastClassChange = System.currentTimeMillis();
276            long lastPomChange = getPomsLastModified();
277    
278            Server server = new Server();
279            if (connectors == null || connectors.length == 0) {
280                SelectChannelConnector selectChannelConnector = new SelectChannelConnector();
281                selectChannelConnector.setPort(8080);
282                connectors = new Connector[]{
283                        selectChannelConnector
284                };
285            }
286            server.setConnectors(connectors);
287            ContextHandlerCollection contexts = new ContextHandlerCollection();
288            HandlerCollection handlerCollection = new HandlerCollection(true);
289            DefaultHandler defaultHandler = new DefaultHandler();
290            handlerCollection.setHandlers(new Handler[]{contexts, defaultHandler});
291            server.setHandler(handlerCollection);
292            try {
293                server.start();
294            } catch (Exception e) {
295                throw new MojoExecutionException(e.getMessage(), e);
296            }
297            List<MavenProject> reactorProjects = this.reactorProjects;
298            WebAppContext webAppContext;
299            Resource webXml;
300            List<Resource> resources;
301            try {
302                resources = new ArrayList<Resource>();
303                for (Artifact a : getOverlayArtifacts(project, scope)) {
304                    addOverlayResources(reactorProjects, resources, a);
305                }
306                if (webAppSourceDirectory == null) {
307                    webAppSourceDirectory = new File(project.getBasedir(), "src/main/webapp");
308                }
309                if (webAppSourceDirectory.isDirectory()) {
310                    resources.add(Resource.newResource(webAppSourceDirectory));
311                }
312                Collections.reverse(resources);
313                getLog().debug("Overlays:");
314                int index = 0;
315                for (Resource r : resources) {
316                    getLog().debug("  [" + index++ + "] = " + r);
317                }
318                final ResourceCollection resourceCollection =
319                        new ResourceCollection(resources.toArray(new Resource[resources.size()]));
320    
321                webAppContext = new WebAppContext();
322                webAppContext.setWar(webAppSourceDirectory.getAbsolutePath());
323                webAppContext.setBaseResource(resourceCollection);
324    
325                WebAppClassLoader classLoader = new WebAppClassLoader(webAppContext);
326                for (String s : getClasspathElements(project, scope)) {
327                    classLoader.addClassPath(s);
328                }
329                webAppContext.setClassLoader(classLoader);
330    
331                contexts.setHandlers(new Handler[]{webAppContext});
332                contexts.start();
333                webAppContext.start();
334                Resource webInf = webAppContext.getWebInf();
335                webXml = webInf != null ? webInf.getResource("web.xml") : null;
336            } catch (MojoExecutionException e) {
337                throw e;
338            } catch (MojoFailureException e) {
339                throw e;
340            } catch (ArtifactFilterException e) {
341                throw new MojoExecutionException(e.getMessage(), e);
342            } catch (MalformedURLException e) {
343                throw new MojoExecutionException(e.getMessage(), e);
344            } catch (IOException e) {
345                throw new MojoExecutionException(e.getMessage(), e);
346            } catch (Exception e) {
347                throw new MojoExecutionException(e.getMessage(), e);
348            }
349    
350            long webXmlLastModified = webXml == null ? 0L : webXml.lastModified();
351            try {
352    
353                getLog().info("Context started. Will restart if changes to poms detected.");
354                long nextClasspathCheck = System.currentTimeMillis() + classpathCheckInterval;
355                while (true) {
356                    long nextCheck = System.currentTimeMillis() + 500;
357                    long pomsLastModified = getPomsLastModified();
358                    boolean pomsChanged = lastPomChange < pomsLastModified;
359                    boolean overlaysChanged = false;
360                    boolean classPathChanged = webXmlLastModified < (webXml == null ? 0L : webXml.lastModified());
361                    if (nextClasspathCheck < System.currentTimeMillis()) {
362                        long classChange = classpathLastModified(project);
363                        if (classChange > lastClassChange) {
364                            classPathChanged = true;
365                            lastClassChange = classChange;
366                        }
367                        nextClasspathCheck = System.currentTimeMillis() + classpathCheckInterval;
368                    }
369                    if (!classPathChanged && !overlaysChanged && !pomsChanged) {
370    
371                        try {
372                            lastResourceChange = processResourceSourceChanges(reactorProjects, project, lastResourceChange);
373                        } catch (ArtifactFilterException e) {
374                            getLog().debug("Couldn't process resource changes", e);
375                        }
376                        try {
377                            Thread.sleep(Math.max(100L, nextCheck - System.currentTimeMillis()));
378                        } catch (InterruptedException e) {
379                            getLog().debug("Interrupted", e);
380                        }
381                        continue;
382                    }
383                    if (pomsChanged) {
384                        getLog().info("Change in poms detected, re-parsing to evaluate impact...");
385                        // we will now process this change,
386                        // so from now on don't re-process
387                        // even if we have issues processing
388                        lastPomChange = pomsLastModified;
389                        List<MavenProject> newReactorProjects;
390                        try {
391                            newReactorProjects = buildReactorProjects();
392                        } catch (ProjectBuildingException e) {
393                            getLog().info("Re-parse aborted due to malformed pom.xml file(s)", e);
394                            continue;
395                        } catch (CycleDetectedException e) {
396                            getLog().info("Re-parse aborted due to dependency cycle in project model", e);
397                            continue;
398                        } catch (DuplicateProjectException e) {
399                            getLog().info("Re-parse aborted due to duplicate projects in project model", e);
400                            continue;
401                        } catch (Exception e) {
402                            getLog().info("Re-parse aborted due a problem that prevented sorting the project model", e);
403                            continue;
404                        }
405                        if (!buildPlanEqual(newReactorProjects, this.reactorProjects)) {
406                            throw new BuildPlanModifiedException("A pom.xml change has impacted the build plan.");
407                        }
408                        MavenProject newProject = findProject(newReactorProjects, this.project);
409                        if (newProject == null) {
410                            throw new BuildPlanModifiedException(
411                                    "A pom.xml change appears to have removed " + this.project.getId()
412                                            + " from the build plan.");
413                        }
414    
415                        newProject.setArtifacts(resolve(newProject, "runtime"));
416    
417                        getLog().debug("Comparing effective classpath of new and old models");
418                        try {
419                            classPathChanged = classPathChanged || classpathsEqual(project, newProject, scope);
420                        } catch (DependencyResolutionRequiredException e) {
421                            getLog().info("Re-parse aborted due to dependency resolution problems", e);
422                            continue;
423                        }
424                        if (classPathChanged) {
425                            getLog().info("Effective classpath of " + project.getId() + " has changed.");
426                        } else {
427                            getLog().debug("Effective classpath is unchanged.");
428                        }
429    
430                        getLog().debug("Comparing effective overlays of new and old models");
431                        try {
432                            overlaysChanged = overlaysEqual(project, newProject);
433                        } catch (OverConstrainedVersionException e) {
434                            getLog().info("Re-parse aborted due to dependency resolution problems", e);
435                            continue;
436                        } catch (ArtifactFilterException e) {
437                            getLog().info("Re-parse aborted due to overlay resolution problems", e);
438                            continue;
439                        }
440                        if (overlaysChanged) {
441                            getLog().info("Overlay modules of " + project.getId() + " have changed.");
442                        } else {
443                            getLog().debug("Overlay modules are unchanged.");
444                        }
445    
446                        getLog().debug("Comparing overlays paths of new and old models");
447                        try {
448                            List<Resource> newResources = new ArrayList<Resource>();
449                            for (Artifact a : getOverlayArtifacts(project, scope)) {
450                                addOverlayResources(newReactorProjects, newResources, a);
451                            }
452                            if (webAppSourceDirectory.isDirectory()) {
453                                newResources.add(Resource.newResource(webAppSourceDirectory));
454                            }
455                            Collections.reverse(newResources);
456                            getLog().debug("New overlays:");
457                            int index = 0;
458                            for (Resource r : newResources) {
459                                getLog().debug("  [" + index++ + "] = " + r);
460                            }
461                            boolean overlayPathsChanged = !resources.equals(newResources);
462                            if (overlayPathsChanged) {
463                                getLog().info("Overlay module paths of " + project.getId() + " have changed.");
464                            } else {
465                                getLog().debug("Overlay module paths are unchanged.");
466                            }
467                            overlaysChanged = overlaysChanged || overlayPathsChanged;
468                        } catch (ArtifactFilterException e) {
469                            getLog().info("Re-parse aborted due to overlay evaluation problems", e);
470                            continue;
471                        } catch (PluginConfigurationException e) {
472                            getLog().info("Re-parse aborted due to overlay evaluation problems", e);
473                            continue;
474                        } catch (PluginContainerException e) {
475                            getLog().info("Re-parse aborted due to overlay evaluation problems", e);
476                            continue;
477                        } catch (IOException e) {
478                            getLog().info("Re-parse aborted due to overlay evaluation problems", e);
479                            continue;
480                        }
481    
482                        project = newProject;
483                        reactorProjects = newReactorProjects;
484                    }
485    
486                    if (!overlaysChanged && !classPathChanged) {
487                        continue;
488                    }
489                    getLog().info("Restarting context to take account of changes...");
490                    try {
491                        webAppContext.stop();
492                    } catch (Exception e) {
493                        throw new MojoExecutionException(e.getMessage(), e);
494                    }
495    
496                    if (classPathChanged) {
497                        getLog().info("Updating classpath...");
498                        try {
499                            WebAppClassLoader classLoader = new WebAppClassLoader(webAppContext);
500                            for (String s : getClasspathElements(project, scope)) {
501                                classLoader.addClassPath(s);
502                            }
503                            webAppContext.setClassLoader(classLoader);
504                        } catch (Exception e) {
505                            throw new MojoExecutionException(e.getMessage(), e);
506                        }
507                    }
508    
509                    if (overlaysChanged || classPathChanged) {
510                        getLog().info("Updating overlays...");
511                        try {
512                            resources = new ArrayList<Resource>();
513                            for (Artifact a : getOverlayArtifacts(project, scope)) {
514                                addOverlayResources(reactorProjects, resources, a);
515                            }
516                            if (webAppSourceDirectory.isDirectory()) {
517                                resources.add(Resource.newResource(webAppSourceDirectory));
518                            }
519                            Collections.reverse(resources);
520                            getLog().debug("Overlays:");
521                            int index = 0;
522                            for (Resource r : resources) {
523                                getLog().debug("  [" + index++ + "] = " + r);
524                            }
525                            final ResourceCollection resourceCollection =
526                                    new ResourceCollection(resources.toArray(new Resource[resources.size()]));
527                            webAppContext.setBaseResource(resourceCollection);
528                        } catch (Exception e) {
529                            throw new MojoExecutionException(e.getMessage(), e);
530                        }
531                    }
532                    try {
533                        webAppContext.start();
534                    } catch (Exception e) {
535                        throw new MojoExecutionException(e.getMessage(), e);
536                    }
537                    webXmlLastModified = webXml == null ? 0L : webXml.lastModified();
538                    getLog().info("Context restarted.");
539                }
540    
541            } finally {
542                try {
543                    server.stop();
544                } catch (Exception e) {
545                    throw new MojoExecutionException(e.getMessage(), e);
546                }
547            }
548        }

550        private void addOverlayResources(List<MavenProject> reactorProjects, List<Resource> resources, Artifact a)
551                throws
552                PluginConfigurationException, PluginContainerException, IOException, MojoExecutionException {
553            MavenProject fromReactor = findProject(reactorProjects, a);
554            if (fromReactor != null) {
555                MavenSession session = this.session.clone();
556                session.setCurrentProject(fromReactor);
557                Plugin plugin = findThisPluginInProject(fromReactor);
558    
559                // we cheat here and use our version of the plugin... but this is less of a cheat than the only
560                // other way which is via reflection.
561                MojoDescriptor jszipDescriptor = findMojoDescriptor(pluginDescriptor, JSZipMojo.class);
562    
563                for (PluginExecution pluginExecution : plugin.getExecutions()) {
564                    if (!pluginExecution.getGoals().contains(jszipDescriptor.getGoal())) {
565                        continue;
566                    }
567                    MojoExecution mojoExecution =
568                            createMojoExecution(plugin, pluginExecution, jszipDescriptor);
569                    JSZipMojo mojo = (JSZipMojo) mavenPluginManager
570                            .getConfiguredMojo(Mojo.class, session, mojoExecution);
571                    try {
572                        File contentDirectory = mojo.getContentDirectory();
573                        if (contentDirectory.isDirectory()) {
574                            getLog().debug(
575                                    "Adding resource directory " + contentDirectory);
576                            resources.add(Resource.newResource(contentDirectory));
577                        }
578                        // TODO filtering support
579                        //
580                        // The good news:
581                        //  * resources:resources gets the list of resources from /project/build/resources *only*
582                        // The bad news:
583                        //  * looks like maven-invoker is the only way to safely invoke it again
584                        //
585                        // probable solution
586                        //
587                        // 1. get the list of all resource directories, add on the scan for changes
588                        // 2. if a change to a non-filtered file, just copy it over
589                        // 3. if a change to a filtered file or a change to effective pom, use maven-invoker to run the
590                        //    lifecycle up to 'compile' or 'process-resources' <-- preferred
591                        //
592                        File resourcesDirectory = mojo.getResourcesDirectory();
593                        if (resourcesDirectory.isDirectory()) {
594                            getLog().debug(
595                                    "Adding resource directory " + resourcesDirectory);
596                            resources.add(Resource.newResource(resourcesDirectory));
597                        }
598                    } finally {
599                        mavenPluginManager.releaseMojo(mojo, mojoExecution);
600                    }
601                }
602            } else {
603                resources.add(Resource.newResource("jar:" + a.getFile().toURI().toURL() + "!/"));
604            }
605        }

607        private void injectMissingArtifacts(MavenProject destination, MavenProject source) {
608            if (destination.getArtifact().getFile() == null && source.getArtifact().getFile() != null) {
609                getLog().info("Pushing primary artifact from forked execution into current execution");
610                destination.getArtifact().setFile(source.getArtifact().getFile());
611            }
612            for (Artifact executedArtifact : source.getAttachedArtifacts()) {
613                String executedArtifactId =
614                        (executedArtifact.getClassifier() == null ? "." : "-" + executedArtifact.getClassifier() + ".")
615                                + executedArtifact.getType();
616                if (StringUtils.equals(executedArtifact.getGroupId(), destination.getGroupId())
617                        && StringUtils.equals(executedArtifact.getArtifactId(), destination.getArtifactId())
618                        && StringUtils.equals(executedArtifact.getVersion(), destination.getVersion())) {
619                    boolean found = false;
620                    for (Artifact artifact : destination.getAttachedArtifacts()) {
621                        if (StringUtils.equals(artifact.getGroupId(), destination.getGroupId())
622                                && StringUtils.equals(artifact.getArtifactId(), destination.getArtifactId())
623                                && StringUtils.equals(artifact.getVersion(), destination.getVersion())
624                                && StringUtils.equals(artifact.getClassifier(), executedArtifact.getClassifier())
625                                && StringUtils.equals(artifact.getType(), executedArtifact.getType())) {
626                            if (artifact.getFile() == null) {
627                                getLog().info("Pushing " + executedArtifactId
628                                        + " artifact from forked execution into current execution");
629                                artifact.setFile(executedArtifact.getFile());
630                            }
631                            found = true;
632                        }
633                    }
634                    if (!found) {
635                        getLog().info("Attaching " +
636                                executedArtifactId
637                                + " artifact from forked execution into current execution");
638                        projectHelper
639                                .attachArtifact(destination, executedArtifact.getType(), executedArtifact.getClassifier(),
640                                        executedArtifact.getFile());
641                    }
642                }
643            }

646        private long processResourceSourceChanges(List<MavenProject> reactorProjects, MavenProject project,
647                                                  long lastModified)
648                throws ArtifactFilterException {
649            long newLastModified = lastModified;
650            getLog().debug("Last modified for resource sources = " + lastModified);
651    
652            Set<File> checked = new HashSet<File>();
653            for (Artifact a : getOverlayArtifacts(project, scope)) {
654                MavenProject p = findProject(reactorProjects, a);
655                if (p == null || p.getBuild() == null || p.getBuild().getResources() == null) {
656                    continue;
657                }
658                boolean changed = false;
659                boolean changedFiltered = false;
660                for (org.apache.maven.model.Resource r : p.getBuild().getResources()) {
661                    File dir = new File(r.getDirectory());
662                    getLog().debug("Checking last modified for " + dir);
663                    if (checked.contains(dir)) {
664                        continue;
665                    }
666                    checked.add(dir);
667                    long dirLastModified = recursiveLastModified(dir);
668                    if (lastModified < dirLastModified) {
669                        changed = true;
670                        if (r.isFiltering()) changedFiltered = true;
671                    }
672                }
673                if (changedFiltered) {
674                    getLog().info("Detected change in resources of " + ArtifactUtils.versionlessKey(a)+"...");
675                    getLog().debug("Resource filtering is used by project, invoking Maven to handle update");
676                    // need to let Maven handle it as its the only (although slower) safe way to do it right with filters
677                    InvocationRequest request = new DefaultInvocationRequest();
678                    request.setPomFile(p.getFile());
679                    request.setInteractive(false);
680                    request.setRecursive(false);
681                    request.setGoals(Collections.singletonList("process-resources"));
682    
683                    Invoker invoker = new DefaultInvoker();
684                    invoker.setLogger(new MavenProxyLogger());
685                    try {
686                        invoker.execute(request);
687                        newLastModified = System.currentTimeMillis();
688                        getLog().info("Change in resources of " + ArtifactUtils.versionlessKey(a) + " processed");
689                    } catch (MavenInvocationException e) {
690                        getLog().info(e);
691                    }
692                } else if (changed) {
693                    getLog().info("Detected change in resources of " + ArtifactUtils.versionlessKey(a)+"...");
694                    getLog().debug("Resource filtering is not used by project, handling update ourselves");
695                    // can do it fast ourselves
696                    MavenResourcesExecution mavenResourcesExecution = new MavenResourcesExecution(p.getResources(), new File(p.getBuild().getOutputDirectory()), p, p.getProperties().getProperty("project.build.sourceEncoding"), Collections.emptyList(), Collections.emptyList(), session);
697                    try {
698                        mavenResourcesFiltering.filterResources(mavenResourcesExecution);
699                        newLastModified = System.currentTimeMillis();
700                        getLog().info("Change in resources of " + ArtifactUtils.versionlessKey(a) + " processed");
701                    } catch (MavenFilteringException e) {
702                        getLog().info(e);
703                    }
704                }
705            }
706            return newLastModified;
707        }

734        private long classpathLastModified(MavenProject project) {
735            long result = Long.MIN_VALUE;
736            try {
737                for (String element : getClasspathElements(project, scope)) {
738                    File elementFile = new File(element);
739                    result = Math.max(recursiveLastModified(elementFile), result);
740                }
741            } catch (DependencyResolutionRequiredException e) {
742                // ignore
743            }
744            return result;
745    
746        }

748        private long recursiveLastModified(File fileOrDirectory) {
749            long result = Long.MIN_VALUE;
750            if (fileOrDirectory.exists()) {
751                result = Math.max(fileOrDirectory.lastModified(), result);
752                if (fileOrDirectory.isDirectory()) {
753                    Stack<Iterator<File>> stack = new Stack<Iterator<File>>();
754                    stack.push(contentsAsList(fileOrDirectory).iterator());
755                    while (!stack.empty()) {
756                        Iterator<File> i = stack.pop();
757                        while (i.hasNext()) {
758                            File file = i.next();
759                            result = Math.max(file.lastModified(), result);
760                            if (file.isDirectory()) {
761                                stack.push(i);
762                                i = contentsAsList(file).iterator();
763                            }
764                        }
765                    }
766                }
767            }
768            return result;
769        }

771        private static List<File> contentsAsList(File directory) {
772            File[] files = directory.listFiles();
773            return files == null ? Collections.<File>emptyList() : Arrays.asList(files);
774        }

776        private boolean classpathsEqual(MavenProject oldProject, MavenProject newProject, String scope)
777                throws DependencyResolutionRequiredException {
778            int seq = 0;
779            List<String> newCP = getClasspathElements(newProject, scope);
780            List<String> oldCP = getClasspathElements(oldProject, scope);
781            boolean classPathChanged = newCP.size() != oldCP.size();
782            for (Iterator<String> i = newCP.iterator(), j = oldCP.iterator(); i.hasNext() || j.hasNext(); ) {
783                String left = i.hasNext() ? i.next() : "(empty)";
784                String right = j.hasNext() ? j.next() : "(empty)";
785                if (!StringUtils.equals(left, right)) {
786                    getLog().debug("classpath[" + seq + "]");
787                    getLog().debug("  old = " + left);
788                    getLog().debug("  new = " + right);
789                    classPathChanged = true;
790                }
791                seq++;
792            }
793            return classPathChanged;
794        }

796        private MavenProject findProject(List<MavenProject> newReactorProjects, MavenProject oldProject) {
797            final String targetId = oldProject.getId();
798            for (MavenProject newProject : newReactorProjects) {
799                if (targetId.equals(newProject.getId())) {
800                    return newProject;
801                }
802            }
803            return null;
804        }

806        private boolean buildPlanEqual(List<MavenProject> newPlan, List<MavenProject> oldPlan) {
807            if (newPlan.size() != oldPlan.size()) {
808                return false;
809            }
810            int seq = 0;
811            for (Iterator<MavenProject> i = newPlan.iterator(), j = oldPlan.iterator(); i.hasNext() && j.hasNext(); ) {
812                MavenProject left = i.next();
813                MavenProject right = j.next();
814                getLog().debug(
815                        "[" + (seq++) + "] = " + left.equals(right) + (left == right ? " same" : " diff") + " : "
816                                + left.getName() + "[" + left.getDependencies().size() + "], " + right.getName()
817                                + "["
818                                + right.getDependencies().size() + "]");
819                if (!left.equals(right)) {
820                    return false;
821                }
822                if (left.getDependencies().size() != right.getDependencies().size()) {
823                    getLog().info("Dependency tree of " + left.getId() + " has been modified");
824                }
825            }
826            return true;
827        }

829        private boolean overlaysEqual(MavenProject oldProject, MavenProject newProject)
830                throws ArtifactFilterException, OverConstrainedVersionException {
831            boolean overlaysChanged;
832            Set<Artifact> newOA = getOverlayArtifacts(newProject, scope);
833            Set<Artifact> oldOA = getOverlayArtifacts(oldProject, scope);
834            overlaysChanged = newOA.size() != oldOA.size();
835            for (Artifact n : newOA) {
836                boolean found = false;
837                for (Artifact o : oldOA) {
838                    if (StringUtils.equals(n.getArtifactId(), o.getArtifactId()) && StringUtils
839                            .equals(n.getGroupId(), o.getGroupId())) {
840                        if (o.getSelectedVersion().equals(n.getSelectedVersion())) {
841                            found = true;
842                            break;
843                        }
844                    }
845                }
846                if (!found) {
847                    getLog().debug("added overlay artifact: " + n);
848                    overlaysChanged = true;
849                }
850            }
851            for (Artifact o : oldOA) {
852                boolean found = false;
853                for (Artifact n : newOA) {
854                    if (StringUtils.equals(n.getArtifactId(), o.getArtifactId()) && StringUtils
855                            .equals(n.getGroupId(), o.getGroupId())) {
856                        if (o.getSelectedVersion().equals(n.getSelectedVersion())) {
857                            found = true;
858                            break;
859                        }
860                    }
861                }
862                if (!found) {
863                    getLog().debug("removed overlay artifact: " + o);
864                    overlaysChanged = true;
865                }
866            }
867            if (overlaysChanged) {
868                getLog().info("Effective overlays of " + oldProject.getId() + " have changed.");
869            } else {
870                getLog().debug("Effective overlays are unchanged.");
871            }
872            return overlaysChanged;
873        }

876        private Set<Artifact> getOverlayArtifacts(MavenProject project, String scope) throws ArtifactFilterException {
877            FilterArtifacts filter = new FilterArtifacts();
878    
879            filter.addFilter(new ProjectTransitivityFilter(project.getDependencyArtifacts(), false));
880    
881            filter.addFilter(new ScopeFilter(scope, ""));
882    
883            filter.addFilter(new TypeFilter(JSZIP_TYPE, ""));
884    
885            return filter.filter(project.getArtifacts());
886        }
