148        public void execute()
149                throws MojoExecutionException, MojoFailureException {
150            if (runPackages == null || runPackages.length == 0) {
151                runPackages = new String[]{"war"};
152            }
153            if (!Arrays.asList(runPackages).contains(project.getPackaging())) {
154                getLog().info("Skipping JSZip run: module " + ArtifactUtils.versionlessKey(project.getGroupId(),
155                        project.getArtifactId()) + " as not specified in runPackages");
156                return;
157            }
158            if (StringUtils.isNotBlank(runModule)
159                    && !project.getArtifactId().equals(runModule)
160                    && !ArtifactUtils.versionlessKey(project.getGroupId(), project.getArtifactId()).equals(runModule)) {
161                getLog().info("Skipping JSZip run: module " + ArtifactUtils.versionlessKey(project.getGroupId(),
162                        project.getArtifactId()) + " as requested runModule is " + runModule);
163                return;
164            }
165            MavenProject project = this.project;
166            long lastPomChange = getPomsLastModified();
167            getLog().info("Servlet container started. Will restart if changes to poms detected.");
168            while (true) {
169                long pomsLastModified = getPomsLastModified();
170                if (lastPomChange == pomsLastModified) {
171                    try {
172                        Thread.sleep(500);
173                    } catch (InterruptedException e) {
174                        getLog().debug("Interrupted", e);
175                    }
176                } else {
177    
178                    getLog().info("Change in poms detected, re-parsing to evaluate impact...");
179                    lastPomChange = pomsLastModified; // we will now process this change, so from now on don't re-process
180                    // even if we have issues processing
181    
182                    List<MavenProject> newReactorProjects;
183                    try {
184                        newReactorProjects = buildReactorProjects();
185                    } catch (ProjectBuildingException e) {
186                        getLog().error(e.getLocalizedMessage());
187                        getLog().info("Re-parse aborted due to malformed pom.xml file(s)");
188                        continue;
189                    } catch (CycleDetectedException e) {
190                        getLog().error(e.getLocalizedMessage());
191                        getLog().info("Re-parse aborted due to dependency cycle in project model");
192                        continue;
193                    } catch (DuplicateProjectException e) {
194                        getLog().error(e.getLocalizedMessage());
195                        getLog().info("Re-parse aborted due to duplicate projects in project model");
196                        continue;
197                    } catch (Exception e) {
198                        getLog().error(e.getLocalizedMessage());
199                        getLog().info("Re-parse aborted due a problem that prevented sorting the project model");
200                        continue;
201                    }
202                    if (!buildPlanEqual(newReactorProjects, this.reactorProjects)) {
203                        throw new BuildPlanModifiedException("A pom.xml change has impacted the build plan.");
204                    }
205                    MavenProject newProject = findProject(newReactorProjects, this.project);
206                    if (newProject == null) {
207                        throw new BuildPlanModifiedException(
208                                "A pom.xml change appears to have removed " + this.project.getId()
209                                        + " from the build plan.");
210                    }
211    
212                    newProject.setArtifacts(resolve(newProject, "runtime"));
213    
214                    getLog().info("Comparing effective classpath of new and old models");
215                    boolean classPathChanged;
216                    try {
217                        classPathChanged = classpathsEqual(project, newProject, scope);
218                    } catch (DependencyResolutionRequiredException e) {
219                        getLog().error(e.getLocalizedMessage());
220                        getLog().info("Re-parse aborted due to dependency resolution problems");
221                        continue;
222                    }
223                    if (classPathChanged) {
224                        getLog().info("Effective classpath of " + project.getId() + " has changed.");
225                    } else {
226                        getLog().info("Effective classpath is unchanged.");
227                    }
228    
229                    getLog().debug("Comparing effective overlays of new and old models");
230                    boolean overlaysChanged;
231                    try {
232                        overlaysChanged = overlaysEqual(project, newProject);
233                    } catch (OverConstrainedVersionException e) {
234                        getLog().error(e.getLocalizedMessage());
235                        getLog().info("Re-parse aborted due to dependency resolution problems");
236                        continue;
237                    } catch (ArtifactFilterException e) {
238                        getLog().error(e.getLocalizedMessage());
239                        getLog().info("Re-parse aborted due to overlay resolution problems");
240                        continue;
241                    }
242                    if (overlaysChanged || classPathChanged) {
243                        getLog().info("Restarting servlet container to take account of changes...");
244                    }
245    
246                    getLog().info("");
247    
248                    project = newProject;
249                }
250    
251            }
252        }
253    
254        private List<MavenProject> buildReactorProjects() throws Exception {
255            List<MavenProject> projects = new ArrayList<MavenProject>();
256            DefaultProfileManager profileManager = new DefaultProfileManager(session.getContainer(),
257                    session.getExecutionProperties());
258            for (MavenProject p : reactorProjects) {
259                projects.add(projectBuilder.build(p.getFile(), localRepository, profileManager));
260            }
261            return new ProjectSorter(projects).getSortedProjects();
262        }

264        private boolean classpathsEqual(MavenProject oldProject, MavenProject newProject, String scope)
265                throws DependencyResolutionRequiredException {
266            int seq = 0;
267            List<String> newCP = getClasspathElements(newProject, scope);
268            List<String> oldCP = getClasspathElements(oldProject, scope);
269            boolean classPathChanged = newCP.size() != oldCP.size();
270            for (Iterator<String> i = newCP.iterator(), j = oldCP.iterator(); i.hasNext() || j.hasNext(); ) {
271                String left = i.hasNext() ? i.next() : "(empty)";
272                String right = j.hasNext() ? j.next() : "(empty)";
273                if (!StringUtils.equals(left, right)) {
274                    getLog().debug("classpath[" + seq + "]");
275                    getLog().debug("  old = " + left);
276                    getLog().debug("  new = " + right);
277                    classPathChanged = true;
278                }
279                seq++;
280            }
281            return classPathChanged;
282        }

284        private MavenProject findProject(List<MavenProject> newReactorProjects, MavenProject oldProject) {
285            final String targetId = oldProject.getId();
286            for (MavenProject newProject : newReactorProjects) {
287                if (targetId.equals(newProject.getId())) {
288                    return newProject;
289                }
290            }
291            return null;
292        }
293    
294        private boolean buildPlanEqual(List<MavenProject> newPlan, List<MavenProject> oldPlan) {
295            if (newPlan.size() != oldPlan.size()) return false;
296            int seq = 0;
297            for (Iterator<MavenProject> i = newPlan.iterator(), j = oldPlan.iterator(); i.hasNext() && j.hasNext(); ) {
298                MavenProject left = i.next();
299                MavenProject right = j.next();
300                getLog().debug(
301                        "[" + (seq++) + "] = " + left.equals(right) + (left == right ? " same" : " diff") + " : "
302                                + left.getName() + "[" + left.getDependencies().size() + "], " + right.getName()
303                                + "["
304                                + right.getDependencies().size() + "]");
305                if (!left.equals(right)) {
306                    return false;
307                }
308                if (left.getDependencies().size() != right.getDependencies().size()) {
309                    getLog().info("Dependency tree of " + left.getId() + " has been modified");
310                }
311            }
312            return true;
313        }

315        private boolean overlaysEqual(MavenProject oldProject, MavenProject newProject)
316                throws ArtifactFilterException, OverConstrainedVersionException {
317            boolean overlaysChanged = false;
318            Set<Artifact> newOA = getOverlayArtifacts(newProject, scope);
319            Set<Artifact> oldOA = getOverlayArtifacts(oldProject, scope);
320            overlaysChanged = newOA.size() != oldOA.size();
321            for (Artifact n : newOA) {
322                boolean found = false;
323                for (Artifact o : oldOA) {
324                    if (StringUtils.equals(n.getArtifactId(), o.getArtifactId()) && StringUtils
325                            .equals(n.getGroupId(), o.getGroupId())) {
326                        if (o.getSelectedVersion().equals(n.getSelectedVersion())) {
327                            found = true;
328                            break;
329                        }
330                    }
331                }
332                if (!found) {
333                    getLog().debug("added overlay artifact: " + n);
334                    overlaysChanged = true;
335                }
336            }
337            for (Artifact o : oldOA) {
338                boolean found = false;
339                for (Artifact n : newOA) {
340                    if (StringUtils.equals(n.getArtifactId(), o.getArtifactId()) && StringUtils
341                            .equals(n.getGroupId(), o.getGroupId())) {
342                        if (o.getSelectedVersion().equals(n.getSelectedVersion())) {
343                            found = true;
344                            break;
345                        }
346                    }
347                }
348                if (!found) {
349                    getLog().debug("removed overlay artifact: " + o);
350                    overlaysChanged = true;
351                }
352            }
353            if (overlaysChanged) {
354                getLog().info("Effective overlays of " + oldProject.getId() + " have changed.");
355            } else {
356                getLog().debug("Effective overlays are unchanged.");
357            }
358            return overlaysChanged;
359        }

361        @SuppressWarnings("unchecked")
362        private Set<Artifact> getOverlayArtifacts(MavenProject project, String scope) throws ArtifactFilterException {
363            FilterArtifacts filter = new FilterArtifacts();
364    
365            filter.addFilter(new ProjectTransitivityFilter(project.getDependencyArtifacts(), false));
366    
367            filter.addFilter(new ScopeFilter(scope, ""));
368    
369            filter.addFilter(new TypeFilter("jszip", ""));
370    
371            return filter.filter(project.getArtifacts());
372        }

374        private long getPomsLastModified() {
375            long result = Long.MIN_VALUE;
376            for (MavenProject p : reactorProjects) {
377                result = Math.max(p.getFile().lastModified(), result);
378            }
379            return result;
380        }

382        @SuppressWarnings("unchecked")
383        private List<String> getClasspathElements(MavenProject project, String scope)
384                throws DependencyResolutionRequiredException {
385            if ("test".equals(scope)) {
386                return (List<String>) project.getTestClasspathElements();
387            }
388            if ("compile".equals(scope)) {
389                return (List<String>) project.getCompileClasspathElements();
390            }
391            if ("runtime".equals(scope)) {
392                return (List<String>) project.getRuntimeClasspathElements();
393            }
394            return Collections.emptyList();
395        }

397        @SuppressWarnings("unchecked")
398        private Set<Artifact> resolve(MavenProject newProject, String scope)
399                throws MojoExecutionException {
400            if (projectDependenciesResolver == null) {
401                getLog().debug("Using Maven 2 strategy to resolve project dependencies");
402                Set<Artifact> resolvedArtifacts;
403                try {
404                    ArtifactResolutionResult result =
405                            artifactResolver.resolveTransitively(newProject.getDependencyArtifacts(),
406                                    artifactFactory.createBuildArtifact(newProject.getGroupId(),
407                                            newProject.getArtifactId(),
408                                            newProject.getVersion(), newProject.getPackaging()),
409                                    newProject.getManagedVersionMap(),
410                                    session.getLocalRepository(),
411                                    newProject.getRemoteArtifactRepositories(),
412                                    metadataSource,
413                                    new ScopeArtifactFilter(scope));
414                    resolvedArtifacts = (Set<Artifact>) result.getArtifacts();
415                } catch (MultipleArtifactsNotFoundException me) {
416                    throw new MojoExecutionException(me.getMessage(), me);
417                } catch (ArtifactNotFoundException e) {
418                    throw new MojoExecutionException(e.getMessage(), e);
419                } catch (ArtifactResolutionException e) {
420                    throw new MojoExecutionException(e.getMessage(), e);
421                }
422                return resolvedArtifacts;
423            } else {
424                getLog().debug("Using Maven e strategy to resolve project dependencies");
425    
426                try {
427                    return (Set<Artifact>) projectDependenciesResolver.getClass()
428                            .getMethod("resolve", MavenProject.class, Collection.class, MavenSession.class)
429                            .invoke(projectDependenciesResolver, newProject, Collections.singletonList(scope), session);
430                } catch (IllegalAccessException e) {
431                    throw new MojoExecutionException(e.getMessage(), e);
432                } catch (InvocationTargetException e) {
433                    throw new MojoExecutionException(e.getMessage(), e);
434                } catch (NoSuchMethodException e) {
435                    throw new MojoExecutionException(e.getMessage(), e);
436                }
437            }
438        }
