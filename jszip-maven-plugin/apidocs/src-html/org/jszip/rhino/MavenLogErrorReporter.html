017    package org.jszip.rhino;
018    
019    import org.apache.maven.plugin.logging.Log;
020    import org.mozilla.javascript.ErrorReporter;
021    import org.mozilla.javascript.EvaluatorException;
022    
023    /**
024     * An {@link ErrorReporter} that routes to a {@link Log}
025     */
026    public class MavenLogErrorReporter implements ErrorReporter {
027    
028        private final Log log;
029    
030        public MavenLogErrorReporter(Log log) {
031            this.log = log;
032        }
033    
034        public void warning(String message, String sourceName, int line, String lineSource,
035                            int lineOffset) {
036            log.warn(message);
037        }
038    
039        public void error(String message, String sourceName, int line, String lineSource,
040                          int lineOffset) {
041            log.error(message);
042        }
043    
044        public EvaluatorException runtimeError(String message, String sourceName, int line,
045                                               String lineSource, int lineOffset) {
046            return new EvaluatorException(message, sourceName, line, lineSource, lineOffset);
047        }
048    }
