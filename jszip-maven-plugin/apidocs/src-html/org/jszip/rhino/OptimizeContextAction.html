048    /**
049     * An action for running r.js against a virtual filesystem.
050     */
051    public class OptimizeContextAction extends ScriptableObject implements ContextAction {
052        private final Global global;
053        private final File profileJs;
054        private final String source;
055        private final int lineNo;
056        private final PseudoFileSystem.Layer[] layers;
057        private final Log log;
058    
059        public OptimizeContextAction(Log log, Global global, File profileJs, String source, int lineNo,
060                                     PseudoFileSystem.Layer... layers) {
061            this.log = log;
062            this.global = global;
063            this.profileJs = profileJs;
064            this.source = source;
065            this.lineNo = lineNo;
066            this.layers = layers;
067        }
068    
069        public Object run(Context context) {
070            context.setErrorReporter(new MavenLogErrorReporter(log));
071            PseudoFileSystem fileSystem = new PseudoFileSystem(layers);
072            context.putThreadLocal(OptimizeContextAction.class, log);
073            fileSystem.installInContext();
074            try {
075    
076                if (log.isDebugEnabled()) {
077                    log.debug("Virtual filesystem exposed to r.js:");
078                    Stack<Iterator<PseudoFile>> stack = new Stack<Iterator<PseudoFile>>();
079                    stack.push(Arrays.asList(fileSystem.root().listFiles()).iterator());
080                    while (!stack.isEmpty()) {
081                        Iterator<PseudoFile> iterator = stack.pop();
082                        while (iterator.hasNext()) {
083                            PseudoFile f = iterator.next();
084                            if (f.isFile()) {
085                                log.debug("  " + f.getAbsolutePath() + " [file]");
086                            } else {
087                                log.debug("  " + f.getAbsolutePath() + " [dir]");
088                                stack.push(iterator);
089                                iterator = Arrays.asList(f.listFiles()).iterator();
090                            }
091                        }
092                    }
093                }
094    
095                List<String> argsList = new ArrayList<String>();
096                argsList.add("-o");
097                argsList.add("/build/" + profileJs.getName());
098                String appDir = null;
099                String baseUrl = "./";
100                String dir = null;
101                try {
102                    String profile = FileUtils.fileRead(profileJs, "UTF-8");
103                    Scriptable scope = context.newObject(global);
104                    scope.setPrototype(global);
105                    scope.setParentScope(null);
106                    Object parsedProfile = context.evaluateString(scope, profile, profileJs.getName(), 0, null);
107                    if (parsedProfile instanceof Scriptable) {
108                        final Scriptable scriptable = (Scriptable) parsedProfile;
109                        appDir = getStringWithDefault(scriptable, "appDir", null);
110                        baseUrl = getStringWithDefault(scriptable, "baseUrl", "./");
111                        dir = getStringWithDefault(scriptable, "dir", null);
112                    }
113                } catch (IOException e) {
114                    log.debug("Cannot infer profile fixups", e);
115                } catch (JavaScriptException e) {
116                    log.warn("JavaScript exception while parsing " + profileJs.getAbsolutePath() + ": " + e.details());
117                } catch (Throwable e) {
118                    log.warn("Cannot infer if profile needs appDir and dir remapping to virtual directory structure", e);
119                }
120                if (appDir == null) {
121                    argsList.add("appDir=/virtual/");
122                    argsList.add("baseUrl=" + baseUrl);
123                } else if (!appDir.startsWith("/virtual/") && !appDir.equals("/virtual")) {
124                    argsList.add("appDir=/virtual/" + StringUtils.removeEnd(StringUtils.removeStart(appDir, "/"),"/")+"/");
125                    argsList.add("baseUrl=" + baseUrl);
126                }
127                if (dir == null) {
128                    argsList.add("dir=/target/");
129                } else if (!dir.startsWith("/target/") && !dir.equals("/target")) {
130                    argsList.add("dir=/target/" + StringUtils.removeEnd(StringUtils.removeStart(dir, "/"),"/")+"/");
131                }
132    
133                global.defineFunctionProperties(new String[]{"print", "quit"}, OptimizeContextAction.class,
134                        ScriptableObject.DONTENUM);
135    
136                Script script = context.compileString(source, "r.js", lineNo, null);
137                script.getClass();
138    
139                Scriptable argsObj = context.newArray(global, argsList.toArray());
140                global.defineProperty("arguments", argsObj, ScriptableObject.DONTENUM);
141    
142                Scriptable scope = context.newObject(global);
143                scope.setPrototype(global);
144                scope.setParentScope(null);
145    
146                NativeJavaTopPackage $packages = (NativeJavaTopPackage) global.get("Packages");
147                NativeJavaPackage $java = (NativeJavaPackage) $packages.get("java");
148                NativeJavaPackage $java_io = (NativeJavaPackage) $java.get("io");
149    
150                ProxyNativeJavaPackage proxy$java = new ProxyNativeJavaPackage($java);
151                ProxyNativeJavaPackage proxy$java_io = new ProxyNativeJavaPackage($java_io);
152                proxy$java_io.put("File", global, get(global, "Packages." + ProxyPseudoFile.class.getName()));
153                proxy$java_io.put("FileInputStream", global,
154                        get(global, "Packages." + PseudoFileInputStream.class.getName()));
155                proxy$java_io.put("FileOutputStream", global,
156                        get(global, "Packages." + PseudoFileOutputStream.class.getName()));
157                proxy$java.put("io", global, proxy$java_io);
158                global.defineProperty("java", proxy$java, ScriptableObject.DONTENUM);
159    
160                log.info("Applying r.js profile " + profileJs.getPath());
161                log.debug("Executing r.js with arguments: " + StringUtils.join(argsList, " "));
162                context.putThreadLocal(ExitCodeHolder.class, new ExitCodeHolder(0));
163                script.exec(context, scope);
164                ExitCodeHolder result = (ExitCodeHolder) context.getThreadLocal(ExitCodeHolder.class);
165                context.putThreadLocal(ExitCodeHolder.class, null);
166                return result.getExitCode();
167            } finally {
168                fileSystem.removeFromContext();
169                context.putThreadLocal(OptimizeContextAction.class, null);
170            }
171        }
172    
173        private String getStringWithDefault(Scriptable scriptable, String name, String defaultValue) {
174            final Object object = scriptable.get(name, scriptable);
175            if (object instanceof String) {
176                return (String) object;
177            }
178            if (object instanceof UniqueTag) {
179                if (object == UniqueTag.NULL_VALUE) {
180                    return null;
181                }
182                return defaultValue;
183            } else {
184                return object.toString();
185            }
186        }
187    
188        private Object get(Scriptable scope, String name) {
189            Scriptable cur = scope;
190            for (String part : StringUtils.split(name, ".")) {
191                Object next = cur.get(part, scope);
192                if (next instanceof Scriptable) {
193                    cur = (Scriptable) next;
194                } else {
195                    return null;
196                }
197            }
198            return cur;
199        }
200    
201        @Override
202        public String getClassName() {
203            return "global";
204        }
205    
206        /**
207         * Print the string values of its arguments.
208         * <p/>
209         * This method is defined as a JavaScript function. Note that its arguments
210         * are of the "varargs" form, which allows it to handle an arbitrary number
211         * of arguments supplied to the JavaScript function.
212         */
213        public static void print(Context cx, Scriptable thisObj, Object[] args, Function funObj) {
214            StringBuilder builder = new StringBuilder();
215            for (int i = 0; i < args.length; i++) {
216                if (i > 0) {
217                    builder.append(" ");
218                }
219    
220                // Convert the arbitrary JavaScript value into a string form.
221                String s = Context.toString(args[i]);
222    
223                builder.append(s);
224            }
225            Log log = (Log) cx.getThreadLocal(OptimizeContextAction.class);
226            if (log != null) {
227                for (String line : builder.toString().split("(\\r\\n?)|(\\n\\r?)")) {
228                    log.info(line);
229                }
230            }
231        }
232    
233        /**
234         * Print the string values of its arguments.
235         * <p/>
236         * This method is defined as a JavaScript function. Note that its arguments
237         * are of the "varargs" form, which allows it to handle an arbitrary number
238         * of arguments supplied to the JavaScript function.
239         */
240        public static void quit(Context cx, Scriptable thisObj, Object[] args, Function funObj) {
241            final int exitCode = args.length == 0 ? 0 : ScriptRuntime.toInt32(args[0]);
242            final Log log = (Log) cx.getThreadLocal(OptimizeContextAction.class);
243            cx.putThreadLocal(ExitCodeHolder.class, new ExitCodeHolder(exitCode));
244            if (exitCode > 0) {
245                if (log != null) {
246                    log.error("Script exit code = " + exitCode);
247                }
248            } else {
249                if (log != null) {
250                    log.debug("Script exit code = " + exitCode);
251                }
252            }
253    
254        }
255    
256        private static class ExitCodeHolder {
257            private final int exitCode;
258    
259            private ExitCodeHolder(int exitCode) {
260                this.exitCode = exitCode;
261            }
262    
263            public int getExitCode() {
264                return exitCode;
265            }
266    
267        }
