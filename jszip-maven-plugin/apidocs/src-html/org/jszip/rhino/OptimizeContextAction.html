017    package org.jszip.rhino;
018    
019    import org.apache.maven.plugin.logging.Log;
020    import org.codehaus.plexus.util.StringUtils;
021    import org.jszip.pseudo.io.PseudoFile;
022    import org.jszip.pseudo.io.ProxyPseudoFile;
023    import org.jszip.pseudo.io.PseudoFileInputStream;
024    import org.jszip.pseudo.io.PseudoFileOutputStream;
025    import org.jszip.pseudo.io.PseudoFileSystem;
026    import org.mozilla.javascript.Context;
027    import org.mozilla.javascript.ContextAction;
028    import org.mozilla.javascript.Function;
029    import org.mozilla.javascript.NativeJavaPackage;
030    import org.mozilla.javascript.NativeJavaTopPackage;
031    import org.mozilla.javascript.Script;
032    import org.mozilla.javascript.Scriptable;
033    import org.mozilla.javascript.ScriptableObject;
034    import org.mozilla.javascript.tools.shell.Global;
035    
036    import java.io.File;
037    import java.util.Arrays;
038    import java.util.Iterator;
039    import java.util.Stack;
040    
041    /**
042     * An action for running r.js against a virtual filesystem.
043     */
044    public class OptimizeContextAction extends ScriptableObject implements ContextAction {
045        private final Global global;
046        private final File profileJs;
047        private final String source;
048        private final int lineNo;
049        private final PseudoFileSystem.Layer[] layers;
050        private final Log log;
051    
052        public OptimizeContextAction(Log log, Global global, File profileJs, String source, int lineNo,
053                                     PseudoFileSystem.Layer... layers) {
054            this.log = log;
055            this.global = global;
056            this.profileJs = profileJs;
057            this.source = source;
058            this.lineNo = lineNo;
059            this.layers = layers;
060        }
061    
062        public Object run(Context context) {
063            context.setErrorReporter(new MavenLogErrorReporter(log));
064            PseudoFileSystem fileSystem = new PseudoFileSystem(layers);
065            context.putThreadLocal(OptimizeContextAction.class, log);
066            fileSystem.installInContext();
067            try {
068    
069                if (log.isDebugEnabled()) {
070                    log.debug("Virtual filesystem exposed to r.js:");
071                    Stack<Iterator<PseudoFile>> stack = new Stack<Iterator<PseudoFile>>();
072                    stack.push(Arrays.asList(fileSystem.root().listFiles()).iterator());
073                    while (!stack.isEmpty()) {
074                        Iterator<PseudoFile> iterator = stack.pop();
075                        while (iterator.hasNext()) {
076                            PseudoFile f = iterator.next();
077                            if (f.isFile()) {
078                                log.debug("  " + f.getAbsolutePath() + " [file]");
079                            } else {
080                                log.debug("  " + f.getAbsolutePath() + " [dir]");
081                                stack.push(iterator);
082                                iterator = Arrays.asList(f.listFiles()).iterator();
083                            }
084                        }
085                    }
086                }
087    
088                global.defineFunctionProperties(new String[]{"print"}, OptimizeContextAction.class,
089                        ScriptableObject.DONTENUM);
090    
091                Script script = context.compileString(source, "r.js", lineNo, null);
092    
093                Scriptable argsObj = context.newArray(global,
094                        new Object[]{"-o", "/build/" + profileJs.getName()});
095                global.defineProperty("arguments", argsObj, ScriptableObject.DONTENUM);
096    
097                Scriptable scope = context.newObject(global);
098                scope.setPrototype(global);
099                scope.setParentScope(null);
100    
101                NativeJavaTopPackage $packages = (NativeJavaTopPackage) global.get("Packages");
102                NativeJavaPackage $java = (NativeJavaPackage) $packages.get("java");
103                NativeJavaPackage $java_io = (NativeJavaPackage) $java.get("io");
104    
105                ProxyNativeJavaPackage proxy$java = new ProxyNativeJavaPackage($java);
106                ProxyNativeJavaPackage proxy$java_io = new ProxyNativeJavaPackage($java_io);
107                proxy$java_io.put("File", global, get(global, "Packages." + ProxyPseudoFile.class.getName()));
108                proxy$java_io.put("FileInputStream", global,
109                        get(global, "Packages." + PseudoFileInputStream.class.getName()));
110                proxy$java_io.put("FileOutputStream", global,
111                        get(global, "Packages." + PseudoFileOutputStream.class.getName()));
112                proxy$java.put("io", global, proxy$java_io);
113                global.defineProperty("java", proxy$java, ScriptableObject.DONTENUM);
114    
115                if (script != null) {
116                    log.info("Applying r.js profile " + profileJs.getPath());
117                    return script.exec(context, scope);
118                }
119                return null;
120            } finally {
121                fileSystem.removeFromContext();
122                context.putThreadLocal(OptimizeContextAction.class, null);
123            }
124        }
125    
126        private Object get(Scriptable scope, String name) {
127            Scriptable cur = scope;
128            for (String part : StringUtils.split(name, ".")) {
129                Object next = cur.get(part, scope);
130                if (next instanceof Scriptable) {
131                    cur = (Scriptable) next;
132                } else {
133                    return null;
134                }
135            }
136            return cur;
137        }
138    
139        @Override
140        public String getClassName() {
141            return "global";
142        }
143    
144        /**
145         * Print the string values of its arguments.
146         * <p/>
147         * This method is defined as a JavaScript function. Note that its arguments
148         * are of the "varargs" form, which allows it to handle an arbitrary number
149         * of arguments supplied to the JavaScript function.
150         */
151        public static void print(Context cx, Scriptable thisObj, Object[] args, Function funObj) {
152            StringBuilder builder = new StringBuilder();
153            for (int i = 0; i < args.length; i++) {
154                if (i > 0) {
155                    builder.append(" ");
156                }
157    
158                // Convert the arbitrary JavaScript value into a string form.
159                String s = Context.toString(args[i]);
160    
161                builder.append(s);
162            }
163            Log log = (Log) cx.getThreadLocal(OptimizeContextAction.class);
164            if (log != null) {
165                for (String line : builder.toString().split("(\\r\\n?)|(\\n\\r?)")) {
166                    log.info(line);
167                }
168            }
169        }
